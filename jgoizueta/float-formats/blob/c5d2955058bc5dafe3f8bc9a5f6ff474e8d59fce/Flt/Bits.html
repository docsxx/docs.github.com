def self.from_s(txt,base=2,len=nil)
  txt = txt.tr('., _','')
  v = txt.to_i(base)
  if len.nil?
    n = power_of_two(base)
    if n
      len = txt.size * n
    end
  end
  from_i v, len
end

def []=(i,b)
  b = (b && b!=0) ? 1 : 0
  @v &= ~(1 << i)
  @v |=  (b << i)
  b
end

def to_s(base=2)
  n = Bits.power_of_two(base)
  if n
    fmt = Nio::Fmt.default.base(base,true).mode(:fix,:exact)
    digits = (size+n-1)/n
    fmt.pad0s!(digits)
    @v.nio_write(fmt)
  else
    @v.to_s(base)
  end
end
