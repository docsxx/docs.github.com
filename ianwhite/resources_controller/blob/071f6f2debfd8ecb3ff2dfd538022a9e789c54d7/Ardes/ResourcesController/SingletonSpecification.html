def initialize(spec_name, options = ;{}, &block)
  options[:segment] ||= spec_name.to_s
  options[:source]  ||= spec_name.to_s
  options[:class]   ||= (options[:source] || spec_name).to_s.camelize.constantize
  super(spec_name, options, &block)
end
