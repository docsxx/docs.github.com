def form_remote_tag(options = {}, &block)
  options[:url].merge!(:protocol => 'https://') if options[:url].kind_of?(Hash)
  options[:html].merge!(:protocol => 'https://') if !options[:html].nil? && options[:html].kind_of?(Hash)

  super(options, &block)
end
