def sites
  return @sites unless @sites.nil?
  @sites = []
  h = JSON.parse(Gosen::Session.session[@uri + "?depth=2"].get(:accept => 'application/json'))['sites']
  h.each do |name, data|
    @sites.push(Gosen::Site.new(data))
  end
  return @sites.sort! { |a,b| a.name <=> b.name }
end
