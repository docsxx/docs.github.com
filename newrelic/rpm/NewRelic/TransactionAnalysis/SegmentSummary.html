def <<(segment)
  if metric_name != segment.metric_name
    raise ArgumentError, "Metric Name Mismatch: #{segment.metric_name} != #{metric_name}" 
  end
  
  @total_time += segment.duration
  @exclusive_time += segment.exclusive_duration
  @call_count += 1
end
