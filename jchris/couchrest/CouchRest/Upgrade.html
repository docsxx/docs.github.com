def clone!
  puts "#{dbname} - #{olddb.info['doc_count']} docs"
  streamer  = CouchRest::Streamer.new(olddb)
  streamer.view("_all_docs_by_seq") do |row|
    load_row_docs(row) if row
    maybe_flush_bulks
  end
  flush_bulks!
end
