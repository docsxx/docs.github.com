def compile(key = nil, options = {})
  method = []
  method << "halted = false"
  each do |callback|
    method << callback.start(key, options)
  end
  method << "yield self if block_given?"
  reverse_each do |callback|
    method << callback.end(key, options)
  end
  method.compact.join("\n")
end
