def to_s
  packet_data = [0xFF, 0xFF, 0xFF].pack("ccc")
  
  unless @split_packet
    packet_data << [0xFF].pack("c")
  else
    packet_data << [0xFE].pack("c")
  end
  
  return packet_data << [@header_data, @content_data.array].pack("ca*")
end
