def initialize(map_data)
  @best_time = map_data.elements['besttimeseconds'].text.to_f
  @id = map_data.name
  @name = map_data.elements['name'].text
  @times_played = map_data.elements['timesplayed'].text.to_i

  case map_data.elements['medal'].text
    when 'gold':
      @medal = L4DMap::GOLD
    when 'silver':
      @medal = L4DMap::SILVER
    when 'bronze':
      @medal = L4DMap::BRONZE
    else
      @medal = L4DMap::NONE
  end
end
