def to_ri_cal_date_time_value(timezone_finder = nil) #:nodoc:
  RiCal::PropertyValue::DateTime.new(
       timezone_finder, 
       :value => strftime("%Y%m%dT%H%M%S"), 
       :params => {"TZID" => self.tzid || :default})
end
