def self.convert(parent, string) #:nodoc:
  ical_str = string.gsub(/\n\r?|\r\n?|,|;|\\/) {|match|
    if ["\n", "\r", "\n\r", "\r\n"].include?(match)
      '\\n'
    else
      "\\#{match}"
    end
    }
  self.new(parent, :value => ical_str)
end

def ruby_value
  if value
    value.gsub(/\\[;,nN\\]/) {|match|
      case match[1,1]
      when /[,;\\]/
        match[1,1]
      when 'n', 'N'
        "\n"
      else
        match
      end
    }
  else
    nil
  end
end
