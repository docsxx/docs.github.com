def method_missing(selector, *args, &b) #:nodoc:
  xprop_candidate = selector.to_s
  if (match = /^x_(.+)(=?)$/.match(xprop_candidate))
    if match[2] == "="
      add_x_property("x_#{match[1]}", *args)
    else
      x_properties[xprop_candidate]
    end
  else
    super
  end
end
