def value=(string)
  @value = string
  parse_match = /([+-])(\d\d)(\d\d)(\d\d)?/.match(string)
  if parse_match
    @sign = parse_match[1] == "+" ? 1 : -1
    @hours = parse_match[2].to_i
    @minutes = parse_match[3].to_i
    @seconds = parse_match[4].to_i || 0
  end
end
