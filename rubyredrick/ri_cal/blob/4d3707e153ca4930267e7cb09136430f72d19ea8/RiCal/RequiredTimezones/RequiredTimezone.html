def add_datetime(date_time)
  if @first_time 
    @first_time = date_time if date_time < @first_time
  else
    @first_time = date_time
  end
  if @last_time 
    @last_time = date_time if date_time > @last_time
  else
    @last_time = date_time
  end
end
