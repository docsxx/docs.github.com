def initialize(main, *args)
  @main      = main
  @options   = args.pop if args.last.is_a?(Hash)
  @resources = args
  @formats   = options[:formats] || :xml

  build!
end

def build!
  resources.each do |r|
    [*formats].each do |f|
      [:member, :collection].each do |t|
        [*options[t]].each do |v|
          send(:"define_#{t}_#{v}", r, f) unless v.nil?
        end
      end
    end
  end
end
