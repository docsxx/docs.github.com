def method_missing(meth, *args, &blk)
  assert_exists

  meth = selector_to_attribute(meth)

  if self.class::ATTRIBUTES.include?(meth) || (self.class == Element && @object.hasAttribute(meth.to_s))
    return @object.getAttribute(meth.to_s)
  end
  Log.warn "Element\#method_missing calling super with #{meth.inspect}"
  super
end

def child_cell(index)
  assert_exists

  if (index - Celerity.index_offset) >= @cells.length
    raise UnknownCellException, "Unable to locate a cell at index #{index}"
  end

  TableCell.new(self, :object, @cells[index - Celerity.index_offset])
end

def locate
  super
  if @object # cant call assert_exists here, as an exists? method call will fail
    @rows = @object.getRows
    @cells = []
    @rows.each do |row|
      row.getCells.each do |c|
        @cells << c
      end
    end
  end

  @object
end
