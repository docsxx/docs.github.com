def complex_expression_sql(op, args)
  case op
  when :'||'
    super(:+, args)
  else
    super(op, args)
  end
end

def multi_insert_sql(columns, values)
  values = values.map {|r| "SELECT #{expression_list(r)}" }.join(" UNION ALL ")
  ["INSERT INTO #{source_list(@opts[:from])} (#{identifier_list(columns)}) #{values}"]
end
