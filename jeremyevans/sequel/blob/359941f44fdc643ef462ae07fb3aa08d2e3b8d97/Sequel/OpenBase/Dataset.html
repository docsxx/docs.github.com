def fetch_rows(sql)
  execute(sql) do |result|
    begin
      @columns = result.column_infos.map{|c| output_identifier(c.name)}
      result.each do |r|
        row = {}
        r.each_with_index {|v, i| row[@columns[i]] = v}
        yield row
      end
    ensure
      # result.close
    end
  end
  self
end
