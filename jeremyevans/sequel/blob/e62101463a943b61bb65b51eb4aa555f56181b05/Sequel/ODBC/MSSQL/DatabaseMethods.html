def execute_insert(sql, opts={})
  log_info(sql)
  synchronize(opts[:server]) do |conn|
    begin
      conn.do(sql)
      log_info(LAST_INSERT_ID_SQL)
      begin
        s = conn.run(LAST_INSERT_ID_SQL)
        if (rows = s.fetch_all) and (row = rows.first)
          Integer(row.first)
        end
      ensure
        s.drop if s
      end
    rescue ::ODBC::Error => e
      raise_error(e)
    end
  end
end
