def initialize(url)
  query = [ "api_key=" + CGI.escape(DigitalNZ.api_key) ]
  url += ".json?" + query * "&"
  puts url
  res = fetch(url)
  res = JSON.parse(res.body)
  puts res.to_yaml
end
