def self.find(*args)
  api = args.last.is_a?(Api) ? args.pop : ANONYMOUS_API
  repo = args.pop
  user = args.pop
  
  user = user.login if user.is_a? User
  if repo.is_a? Repository
    repo = repo.name 
    user ||= repo.owner 
  end
  
  self.validate_args(user => :user, repo => :repo)
  super user, repo, api
end
