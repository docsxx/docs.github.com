def fetch_remaining
  unless @fully_fetched
    columns_to_fetch = @klass.column_names - @scrooge_columns.to_a
    unless columns_to_fetch.empty?
      fetch_remaining!( columns_to_fetch )
    end
    @fully_fetched = true
  end
end
