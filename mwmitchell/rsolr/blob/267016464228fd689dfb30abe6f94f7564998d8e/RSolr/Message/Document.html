def initialize(doc_hash = {})
  @fields = []
  doc_hash.each_pair do |field,values|
    # create a new field for each value (multi-valued)
    # put non-array values into an array
    values = [values] unless values.is_a?(Array)
    values.each do |v|
      next if v.to_s.empty?
      @fields << Field.new({:name=>field}, v)
    end
  end
  @attrs={}
end

Add a field value to the document. Options map directly to XML attributes in the Solr <field> node. See http://wiki.apache.org/solr/UpdateXmlMessages#head-8315b8028923d028950ff750a57ee22cbf7977c6

def add_field(name, value, options = {})
  @fields << Field.new(options.merge({:name=>name}), value)
end
