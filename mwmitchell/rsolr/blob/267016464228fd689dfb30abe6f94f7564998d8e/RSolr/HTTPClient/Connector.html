def connect(url)
  case adapter_name
  when :curb
    klass = 'Curb'
  when :net_http
    klass = 'NetHTTP'
  else
    raise UnkownAdapterError.new("Name: #{adapter_name}")
  end
  begin
    RSolr::HTTPClient::Base.new RSolr::HTTPClient::Adapter.const_get(klass).new(url)
  rescue ::URI::InvalidURIError
    raise "#{$!} == #{url}"
  end
end
