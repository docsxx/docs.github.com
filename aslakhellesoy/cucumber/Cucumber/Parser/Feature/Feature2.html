def build(filter)
  if(filter.nil? || feature_elements.accept?(filter) || (!bg.empty? && filter.accept?(bg)))
    background = bg.respond_to?(:build) ? bg.build : nil      
    Ast::Feature.new(
      background, 
      comment.build, 
      tags.build, 
      header.text_value, 
      feature_elements.build(background, filter)
    )
  end
end
