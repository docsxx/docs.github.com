def raw(filter=nil, scenario_outline=nil)
  elements.map do |table_row|
    if(filter.nil? || table_row == elements[0] || filter.at_line?(table_row) || (scenario_outline && filter.outline_at_line?(scenario_outline)))
      table_row.build
    end
  end.compact
end
