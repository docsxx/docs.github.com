def build(background, filter)
  Ast::ScenarioOutline.new(
    background,
    comment.build, 
    tags.build,
    scenario_outline_keyword.line, 
    scenario_outline_keyword.text_value, 
    name.build, 
    steps.build, 
    examples_sections.build(filter, self)
  )
end
