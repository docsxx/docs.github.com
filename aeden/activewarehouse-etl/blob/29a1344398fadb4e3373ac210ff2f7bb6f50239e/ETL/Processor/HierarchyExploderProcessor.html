def process(row)
  rows = []
  target = configuration[:target]
  table = configuration[:table]
  conn = ETL::Engine.connection(target)
  build_rows([row[:id]], row[:id], row[:id], row[:parent_id].nil?, 0, rows, table, conn)
  rows
end
