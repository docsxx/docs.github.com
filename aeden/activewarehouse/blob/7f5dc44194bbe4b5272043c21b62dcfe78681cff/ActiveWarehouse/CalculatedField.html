def initialize(fact_class, name, type = :integer, field_options = ;{}, &block)
  unless block_given?
    raise ArgumentError, "A block is required for the calculated field #{name} in #{fact_class}"
  end
  super(fact_class, name.to_s, type, field_options)
  @block = block
end
