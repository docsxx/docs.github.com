def view(params, options = ;{})
  if options.has_key?(:sortable_with_totals)
    options[:sortable] = true
    options[:with_totals] = true
    options.delete(:sortable_with_totals)
  end
  
  ActiveWarehouse::View::TableView.new(self, params, options)
end
