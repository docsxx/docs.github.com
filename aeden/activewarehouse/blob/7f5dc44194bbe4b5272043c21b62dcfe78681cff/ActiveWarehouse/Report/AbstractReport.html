def column_hierarchy
  ch = @column_hierarchy
  if ch.nil? || ch == 'NULL'
    column_dimension_class.hierarchies.first
  else
    ch
  end
end

def fact_attributes
  @fact_attributes ||= returning Array.new do |fa|
    fact_class.aggregate_fields.each { |field| fa << field }
    fact_class.calculated_fields.each { |field| fa << field }
  end
end
