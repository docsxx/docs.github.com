def initialize(key=nil, parent_cell=nil)
  @closed = false
  @processed = false
  @parent = parent_cell
  @parent.child = self if @parent
  @index = @@sequence += 1
  #puts "creating node #{@index} with parent: #{@parent}"
  add_cell(Cell.new(key)) if key
end

  end
