def fiscal_year_yday(offset_month=10)
  offset_days = 0
  1.upto(offset_month - 1) { |m| offset_days += ::Time.days_in_month(m, year) }
  shifted_year_day = yday - offset_days
  shifted_year_day += 365 if shifted_year_day <= 0
  shifted_year_day
end
