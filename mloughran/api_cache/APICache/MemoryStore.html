def get(key)
  data = @cache[key][1]
  APICache.logger.debug("cache: #{data.nil? ? "miss" : "hit"} (#{key})")
  data
end

def set(key, value)
  APICache.logger.debug("cache: set (#{key})")
  @cache[key] = [Time.now, value]
  true
end
