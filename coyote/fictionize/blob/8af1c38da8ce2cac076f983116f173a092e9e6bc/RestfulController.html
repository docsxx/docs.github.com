def destroy
  @resource = model_name.find(params[:id])
  @resource.destroy

  respond_to do |format|
    format.html { redirect_to index_url }
    format.xml  { head :ok }
  end
end

def index
  page = params[:page] || 1
  @resources = model_name.paginate :page => page, :per_page => page_size, :order => 'created_at DESC'
  respond_to do |format|
    format.html # index.html.haml
    format.xml  { render :xml => @resources }
  end
end

def update
  @resource = model_name.find(params[:id])
  @resource.save_associated_models(params)     if  @resource.respond_to?(:save_associated_models)
  p = params_for_update(params[model_name.to_s.downcase])
  respond_to do |format|
    if @resource.update_attributes(p)
      flash[:notice] = "#{model_name.to_s}  was successfully updated."
      format.html { redirect_to created_ok_url(params['goto_id']) }
      format.xml  { head :ok }
    else
      format.html { render :action => "edit" }
      format.xml  { render :xml => @model_name.errors, :status => :unprocessable_entity }
    end
  end
end
