Class: Bob::BackgroundEngines::Threaded::ThreadPool

  • Object
    • Bob::BackgroundEngines::Threaded::ThreadPool

Manage a pool of threads, allowing for spin up / spin down of the contained threads. Simply processes work added to it’s queue via #push. The default size for the pool is 2 threads.

def initialize(size = nil)
  size ||= 2

  @njobs   = Incrementor.new
  @workers = Array.new(size) { spawn }
end

def add(*jobs, &blk)
  jobs = jobs + Array(blk)

  jobs.each do |job|
    @jobs << job
    @njobs.inc
  end
end

def size=(other)
  @size = other

  if @workers.size > @size
    (@workers.size - @size).times do
      @workers.shift[:run] = false
    end
  else
    (@size - @workers.size).times do
      @workers << spawn
    end
  end
end
