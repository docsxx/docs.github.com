def add_column(&block)
  column = ColumnBuilder.new(@template)
  yield column
  @columns << column
end

def counted_width
  @columns.inject(0) do |total, column|
    total + column.width
  end
end

def render
  @columns.each do |column|
    column.css_classes << "last" if column == @columns.last
    @html << column.render
  end

  @html
end
