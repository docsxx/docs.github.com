def server
  if @server
    @server
  else
    opts = {:content_type =>'application/json', 
     :accept      => 'application/json',
     :host        => 'http://localhost',
     :port        => '3000'}.merge(server_config)
    @uri = "#{opts.delete(:host)}:#{opts.delete(:port)}"
    @server = RestClient::Resource.new( @uri, opts)
  end
end
