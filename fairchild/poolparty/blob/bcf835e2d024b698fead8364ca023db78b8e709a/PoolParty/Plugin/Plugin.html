def initialize(opts=;{}, extra_opts={}, prnt=nil, &block)        
  setup_callbacks
  
  run_in_context do
    before_load(opts, &block)
  end
  
  block = Proc.new {enable} unless block        
  
  super(opts, &block)
  
  run_in_context do
    loaded opts, &block
  end
  
  after_create
end
