def get_access_key        
  key_help_str = "Now you need to get your keys from Amazon AWS.\nIf you don't already have your keys setup, navigate to http://aws.amazon.com\nand click on Your Account. Click on the Access Identifiers tab and find:\nYour Access Key ID. This is your access key.\n"
  access_key_str = "Since you'll be using ec2, we'll have to set some things up before\nwe can get going on PoolParty. Don't worry, this information will stay between us.\n\n"
  say access_key_str
  ask_with_help :message => "First, what's your access key?",
                :help => key_help_str do |t|
    @access_key = t
  end
end

def get_keypair
  begin
    ec2 = PoolParty::Remote::Ec2.ec2({:access_key => @access_key, :secret_access_key => @secret_access_key})
    keypairs = ec2.describe_keypairs["keySet"]["item"]
    keynames = keypairs.map {|k| k["keyName"]}  
  rescue Exception => e
    colored_say "There was an error: #{e}. Recheck your access_key and secret_access_key to make sure you copied them correctly"
    exit 1
  end

  key_str ="\nFinally, what's the name of the keypair you'd like to use?\n\nYou already have the following keypairs setup:\n\\t\#{keynames.join(\"\\n\\t\")}\n\nYou can use one of these keys, or create a new one.\n"

  key_str_help ="To make a new keypair, make sure you have the ec2 tools installed. You can create a new one by typing the command:\nec2-add-keypair\n\nSave this into a file in your ~/.ec2 directory. Enter the name of the keypair here.\n"
  ask_with_help :message => key_str,
                :help => key_str_help do |t|
      @keypair = t
      say "\nIn your clouds.rb, you can use this keypair in your clouds.\n\nCreate one for each cloud you want to use.\n\n"
  end
end

def show_cert_message
  rescued_ask_str = "Super duper! You'll need to make sure you have yout X.509 certificate downloaded too.\nSave this in your ~/.ec2 directory.\n\nPress enter when you're ready\n"

  cert_help_str = "To get your X.509 certificates, navigate to http://aws.amazon.com and login. Click on\nYour Account and scroll down. The X.509 Certificate box is at the bottom. Make sure you\nclick download and save the cert-*.pem file. If you don't know or don't have the pk-*.pem file\nyou may have to recreate it. Not to worry though, it's super easy. Click on the Create New button.\nFrom there, you can download the cert-*.pem and the pk-*.pem files.\n\n"

def write_clouds_rb
  clds ="pool :my_pool do\ncloud :my_app do\n\n# Copy these lines and source them in your .profile or .bashrc file\n# export AWS_ACCESS_KEY=\"\#{@access_key}\"\n# export AWS_SECRET_ACCESS_KEY=\"\#{@secret_access_key}\"\n# export EC2_PRIVATE_KEY=$(ls ~/.ec2/pk-*.pem)\n# export EC2_CERT=$(ls ~/.ec2/cert-*.pem)\n\naccess_key \"\#{@access_key}\"\nsecret_access_key \"\#{@secret_access_key}\"\nkeypair \"\#{@keypair}\"\n\nhas_file \"/etc/motd\" do\ncontent \"Welcome to your first PoolParty instance!\"\nend\nend\nend\n"

  ::File.open("clouds.rb", "w") {|f| f << clds}
end
