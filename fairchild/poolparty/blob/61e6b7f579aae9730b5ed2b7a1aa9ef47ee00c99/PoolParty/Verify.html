def constantize(name, base_object=Monitors)
  begin
    const = base_object.constants.find{|cnst| cnst == name.camelcase }
    base_object.module_eval const
  rescue Exception => e
    puts "#{name.camelcase} is not defined. #{e}"
    nil
  end
end
