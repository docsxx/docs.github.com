def self.run_cmd(commands, target_host)
  p [:running_commands, commands]
  if target_host || target_host=='localhost'
    out = run_local commands
  else
   out = run_remote host, commands
  end
  raise "FAILED: run_cmd #{commands}" unless $?.success?
  out
end

def describe_instances(o={})
  output = `virsh list`.split("\n")
  return [] if output.empty? || output.size < 3
  output[2..-1].collect do |i|
    d=i.split(' ')
    LibvirtInstance.new(:name => d[1], :status  => d[2..-1].join(' '))
  end
end
