def debug_prompt(cmd)
  ui.say("Preparing to execute command: #{cmd}")
  prompt = "Execute ([Yes], No, Abort) "
  ui.ask("#{prompt}?  ") do |q|
    q.overwrite = false
    q.default = 'y'
    q.validate = /(y(es)?)|(no?)|(a(bort)?|\n)/i
    q.responses[:not_valid] = prompt
  end
end
