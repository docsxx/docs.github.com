def << (job)

  @mutex.synchronize do
    delete(job.job_id)
    @jobs << job
    @jobs.sort! { |j0, j1| j0.at <=> j1.at }
  end
end

def select (type)

  type = JOB_TYPES[type]
  @jobs.select { |j| j.is_a?(type) }
end

def trigger_matching_jobs

  now = Time.now

  while job = job_to_trigger(now)
    job.trigger
  end
end
