def initialize (opts={})

  @frequency = @options[:frequency] || 0.330

def handle_exception (job, exception)

  if self.respond_to?(:log_exception)
    #
    # some kind of backward compatibility

    log_exception(exception)

  else

    puts '=' * 80
    puts "scheduler caught exception :"
    puts exception
    puts '=' * 80
  end
end
