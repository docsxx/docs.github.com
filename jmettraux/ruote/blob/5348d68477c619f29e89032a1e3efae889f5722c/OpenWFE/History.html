def log (source, event, *args)

  t = Time.now

  msg = "#{t} .#{t.usec} -- #{source.to_s} #{event.to_s}"

  msg << " #{get_fei(args).to_s}" if args.length > 0

  m = get_message(source, event, args)
  msg << " #{m}" if m

  @output << msg + "\n"
end
