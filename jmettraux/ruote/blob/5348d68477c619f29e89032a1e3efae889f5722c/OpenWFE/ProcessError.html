Class: OpenWFE::ProcessError

Encapsulating process error information.

Instances of this class may be used to replay_at_error

def to_s
  s = ''
  s << "-- #{self.class.name} --\n"
  s << "   date : #{@date}\n"
  s << "   fdate : #{@fdate}\n"
  s << "   fei : #{@fei}\n"
  s << "   message : #{@message}\n"
  s << "   workitem : ...\n"
  s << "   error_class : #{@error_class}\n"
  s << "   stacktrace : #{@stacktrace[0, 80]}\n"
  s
end
