Class: OpenWFE::DefaultWfidGenerator

  • OpenWFE::Service
    • OpenWFE::DefaultWfidGenerator

This default wfid generator outputs a long integer as a String. The last given id (in order to prevent clocks being put back) is stored in the work directory in the file "wfidgen.last"

def initialize (service_name, application_context)

  super

  @last = -1
  @mutex = Mutex.new

  @last_fn = get_work_directory + '/wfidgen.last'

  load_last

  ensure_last_f
end

def self.split_wfid (wfid)

  r = []
  0.upto(wfid.length-1) do |i|
    r << wfid[i, 1]
  end

  r
end

def generate (launchitem=nil)

  wfid = nil

  @mutex.synchronize do
    wfid = now
    wfid = @last + 1 if wfid <= @last
    @last = wfid
    save_last
  end

  to_string(wfid)
end

def stop
