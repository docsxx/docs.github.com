def deep_symbolize_keys
  output = Hash.new
  self.each do |key, value|
    if value.is_a?(Hash)
      output[key.to_sym] = value.symbolize_keys
    else
      output[key.to_sym] = value
    end
  end
  return output
end

def to_native
  self.each do |key, value|
    value = case value
    when "true" then true
    when "false" then false
    when "nil" then nil
    when /^\d+$/ then value.to_i
    when /^\d+\.\d+$/ then value.to_f
    else value end
    self[key.to_sym] = value
  end
  return self
end
