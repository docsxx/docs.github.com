def method_missing(method, *args)
  if method.to_s.include?("_and_")
    method.to_s.split("_and_").each do |meth|
      self.send(meth, *args)
    end
  elsif self[0].respond_to?(method)
    each { |video| video.send(method, *args) }
  else
    super
  end
end

def make_zip_filename_for(dir)

end
