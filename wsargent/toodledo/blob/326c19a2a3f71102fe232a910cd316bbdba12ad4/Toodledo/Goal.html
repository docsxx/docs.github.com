def self.parse(session, el)      
  id = el.attributes['id']
  level = el.attributes['level'].to_i
  contributes_id = el.attributes['contributes']
  name = el.text
  goal = Goal.new(id, level, contributes_id, name)
  return goal    
end

def self.valid?(input)
  for level in LEVEL_ARRAY
    if (level == input)
      return true
    end
  end
  return false
end
