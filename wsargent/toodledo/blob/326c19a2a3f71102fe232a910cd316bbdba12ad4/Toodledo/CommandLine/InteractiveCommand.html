def command_loop(session)
  loop do
    begin
      input = ask("> ") do |q|
        q.readline = true
      end
      
      input.strip!
      
      client.execute_command(session, input)            
    rescue Toodledo::ItemNotFoundError => infe
      puts "Item not found: #{infe}"
    rescue Toodledo::ServerError => se
      puts "Server Error: #{se}"
    rescue RuntimeError => re
      puts "Error: #{re}"      
    end
  end # loop    

end
