def new_from_xml(xml, doc=nil)
  name    = (xml).at(:name).inner_html
  t       = Tag.new(name)
  t.count = (xml).at(:count).inner_html
  t.url   = (xml).at(:url).inner_html
  t
end

def top_tags
  doc = fetch_and_parse("/#{API_VERSION}/tag/toptags.xml")
  @top_tags = (doc/:tag).inject([]) do |tags, tag|
    t       = Tag.new(tag['name'])
    t.count = tag['count']
    t.url   = tag['url']
    tags << t
    tags
  end
end
