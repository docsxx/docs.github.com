def initialize(attributes = {})
  @distance = 0
  @distance_old = 0
  @time = 0
  @time_old = 0
  @best_time = Infinity
  @wins = 0
  @races = 0
  @speed = 0
  raise unless $ROLLER_CIRCUMFERENCE
  @roller_circumference = $ROLLER_CIRCUMFERENCE
  @ticks = 0
  @name = attributes[:name]
  @color = attributes[:color]
  @race_distance = attributes[:race_distance]
  case UNIT_SYSTEM
    when :mph
      @unit_conversion = METERS_PER_MILLISECOND_TO_MILES_PER_HOUR
    when :kph
      @unit_conversion = METERS_PER_MILLISECOND_TO_KILOMETERS_PER_HOUR
  end
end

def speed(time)
  @distance = self.distance
  @time = time
  if(@time_old > @time)
    @time_old = time
  end
  if time == 0
    0
  else
    if(@time-@time_old > 999)
      if(@distance_old > 0)
        @speed = "%.2f" % (((@distance - @distance_old) / (@time - @time_old)) * @unit_conversion).to_f
      else
        @speed = 0
      end
      @distance_old = @distance
      @time_old = @time
    end
    @speed
  end
end
