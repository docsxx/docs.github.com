def parse_datetime(string)
  begin
    DateTime.parse(string).feed_utils_to_gm_time
  rescue
    puts "DATE CAN'T BE PARSED: #{string}"
    nil
  end
end

def published=(val)
  parsed = parse_datetime(val)
  @published = parsed if !@published || parsed < @published
end
