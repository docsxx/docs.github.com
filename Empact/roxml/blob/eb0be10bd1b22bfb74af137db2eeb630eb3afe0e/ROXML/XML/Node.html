def self.from(data)
  case data
  when XML::Node
    data
  when XML::Document
    data.root
  when File, IO
    Parser.parse_io(data).root
  else
    if (defined?(URI) && data.is_a?(URI::Generic)) ||
       (defined?(Pathname) && data.is_a?(Pathname))
      Parser.parse_file(data.to_s).root
    else
      Parser.parse(data).root
    end
  end
end

def search(xpath)
  begin
    REXML::XPath.match(self, xpath)
  rescue Exception => ex
    raise ex, xpath
  end
end
