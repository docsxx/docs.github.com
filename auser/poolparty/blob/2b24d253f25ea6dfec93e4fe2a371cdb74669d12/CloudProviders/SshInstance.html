def cloud_provider(o={}, &block)
  @cloud_provider ||= if cloud
    cloud.cloud_provider
  else
    options_for_cloud_provider = o.choose{|k,v| Ssh.default_options.has_key?(k)}
    Ssh.new( options_for_cloud_provider, &block)
  end
end
