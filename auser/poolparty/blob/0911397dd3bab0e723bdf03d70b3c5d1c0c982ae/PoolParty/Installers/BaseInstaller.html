def welcome_message
  welcome_msg = "We are going to take you through the installation process of PoolParty.\n\nFirst, we'll setup your environment so using PoolParty will be a breeze\n"
          
  colored_say "Welcome to PoolParty!", :help
  say welcome_msg
  begin
    t = colored_ask "Press enter to continue or Ctrl+C to exit"
  rescue Exception => e
    say "\nCancelled PoolParty installation\n\nYou can always restart this by typing:\ninstall-poolparty\n"
    exit 0
  end
  
end
