def method_missing(m,*a,&block)
  if Default.respond_to?(m)
    Default.send(m,*a,&block)
  else
    super
  end
end

def self.define_resource_methods
  ddputs "Defining resources..."
  defined_resources.each do |res|
    ddputs "Defining resource: #{res}"
    Base.class_eval "def has_\#{res.has_method_name}(a={},b={},&block)\nobj = \#{res}.new(a,b,&block)\nobj.exists!\nordered_resources << obj\nobj\nend\ndef does_not_have_\#{res.has_method_name}(a={},b={},&block)\nobj = has_\#{res.has_method_name}(a,b,&block)\nobj.does_not_exist!\nobj\nend\ndef \#{res.has_method_name}s\nordered_resources.select {|q| q if q.class.to_s =~ /\#{res.to_s.classify}/ }\nend\nalias :\#{res.has_method_name} :has_\#{res.has_method_name}\n\ndef get_\#{res.has_method_name}(nm)\nout_res = \#{res.has_method_name}s.detect {|other| other.base_name == nm}\nout_res ||= current_context.reverse.detect {|s| s.get_\#{res.has_method_name}(nm) if s}\nraise PoolParty::PoolPartyError.create(\"ResourceNotFound\", \"The \#{res.has_method_name} \\\#{nm} was not found. Please make sure you've specified it\") unless out_res\nout_res\nend\n"
  end
end

def print_variable(obj)
  case obj
  when Fixnum
    case obj
    when /^\d{3}$/
      "0#{obj.to_i}"
    else
      "#{obj.to_i}"
    end        
  when String
    case obj
    when /^\d{4}$/
      "#{obj}"
    when /^\d{3}$/
      "0#{obj}"
    else
      "\"#{obj}\""
    end
  when Proc
    obj.call # eh
  when Array
    "[ #{obj.map {|e| print_variable(e) }.reject {|a| a.nil? || a.empty? }.join(", ")} ]"
  when nil
    nil
  when Symbol
    ":#{obj}"
  when Hash
    "#{obj.map {|k,v| ":#{k} => #{print_variable(v)}" unless v == obj }.compact.join(",\n")}"
  else
    "#{obj}"
  end
end
