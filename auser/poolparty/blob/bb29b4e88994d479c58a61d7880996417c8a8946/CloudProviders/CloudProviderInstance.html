def wait_for_port(port, opts={})
  timeout = opts[:timeout] || 1
  ip      = opts[:public_ip] || public_ip
  begin
    Timeout::timeout(timeout) do
      begin
        s = TCPSocket.new(ip, port)
        s.close
        return true
      rescue Errno::ECONNREFUSED, Errno::EHOSTUNREACH
        return false
      end
    end
  rescue Timeout::Error
  end
  return false
end
