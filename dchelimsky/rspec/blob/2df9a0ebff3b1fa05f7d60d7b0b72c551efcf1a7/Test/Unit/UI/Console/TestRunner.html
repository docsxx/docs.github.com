def test_started_with_rspec(name)
  if @need_to_output_started
    if @rspec_io
      @rspec_io.rewind
      output(@rspec_io.read)
    end
    output("Started")
    @need_to_output_started = false
  end
  test_started_without_rspec(name)
end
