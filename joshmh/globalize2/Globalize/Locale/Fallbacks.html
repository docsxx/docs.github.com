def initialize(*defaults)
  @map = {}
  map defaults.pop if defaults.last.is_a?(Hash)
  
  defaults = [I18n.default_locale.to_sym] if defaults.empty?
  self.defaults = defaults 
end

def map(mappings)
  mappings.each do |from, to|
    from, to = from.to_sym, Array(to)
    to.each do |to|
      @map[from] ||= []
      @map[from] << to.to_sym
    end
  end
end
