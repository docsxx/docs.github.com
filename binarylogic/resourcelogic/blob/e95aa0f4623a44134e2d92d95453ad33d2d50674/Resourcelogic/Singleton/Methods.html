def object
  return @object if defined?(@object)
  
  if singleton?
    if !parent? && respond_to?("current_#{model_name}", true)
      @object = send("current_#{model_name}")
    elsif parent? && parent_object.send(model_name)
      @object = parent_object.send(model_name)
    else
      super
    end
  else
    super
  end
end
