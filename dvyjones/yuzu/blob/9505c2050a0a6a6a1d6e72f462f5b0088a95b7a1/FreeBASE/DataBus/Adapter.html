Class: FreeBASE::DataBus::Adapter

The Adapter is a helper class that allows for connecting slot values together (in a Thread-safe way) so that changes to one (or more) slots would result in automatic updates to other slots

def initialize(slots, &block)
  block = proc if proc
  slots.each {|slot| slot.subscribe(self)}
  @proc = block
  @thread_queue = []
end

def databus_notify(msg, slot)
  return if @current_thread == Thread.current
  if @current_thread
    @thread_queue << Thread.current
    Thread.current.stop
  end
  @current_thread = Thread.current
  @proc.call(msg, slot)
  @current_thread = nil
  @thread_queue.shift.wake if @thread_queue.size>0
end
