Class: FreeBASE::SubsystemConfiguration

Represents a subsystem (set of plugins) in FreeBASE. It creates a plugin named the subsystem name which is dependent on all plugins in the subsystem. This allows another plugin to be dependent on all the plugins in the subsystem by simply being dependent on the plugin named after the subsystem

def add_plugin(&block)
  plugin = PluginConfiguration.new(&block)
  plugin.configuration = @configuration
  @plugins << plugin
  plugin
end

def configuration=(configuration)
  @configuration = configuration
  each_plugin {|plugin| plugin.configuration = configuration}
  #add plugin to represent the subsystem itself
  add_plugin do |plugin|
    plugin.name = @name
    plugin.version = @version
    plugin.require_path = nil #"freebase/plugin"
    plugin.startup_module = "FreeBASE::SubsystemPlugin"
    plugin.autoload = true
    @plugins.each do |other|
      plugin.add_load_dependency(other.name)
    end
    @plugins.each do |other|
      plugin.add_start_dependency(other.name)
    end
  end
end
