Class: FreeBASE::DataBus::ProcWrapper

Usage:p = ProcWrapper.new(self) p.set p2| puts p1,p2 p.call("one", "two") #=> "one"\n"two"

def call(*args)
  @slot.validate(args)
  result = nil
  result = @proc.call(*args) if @proc
  @slot.notify(:notify_proc_call)
  return result
end

def set_proc(proc=nil, &block)
  block = proc if proc
  @proc = block
  @slot.notify(:notify_proc_set)
end
