def self.set_main_loop
  bus["/system/ui/messagepump"].set_proc do
    begin
      App.log.info "starting Gui.main (in thread #{Thread.current})"
      Hook.trigger(:redcar_start)
      if in_features_process?
        Redcar::Testing::InternalCucumberRunner.begin_tests
      else
        Toolkit.start
        bus["/system/shutdown"].call(1)
        # exit(0)
      end
    rescue Object => e
      $stderr.puts str="\n---------------------------\nRedcar has crashed.\n---------------------------\n\nPlease report this error message to the Redcar mailing list, along\nwith the actions you were taking at the time:\n\nTime: \#{Time.now}\nMessage: \#{e.message}\nBacktrace: \\n\#{e.backtrace.map{|l| \"    \"+l}.join(\"\\n\")}\nUname -a: \#{`uname -a`.chomp}\n"
    end
  end
end
