def execute(obj, *args)
  case @on_transition
  when Symbol, String
    obj.send(@on_transition, *args)
  when Proc
    @on_transition.call(obj, *args)
  end
end

def perform(obj)
  case @guard
  when Symbol, String
    obj.send(@guard)
  when Proc
    @guard.call(obj)
  else
    true
  end
end
