def send_confirmation_instructions(on=:create)
  self.reset_perishable_token
  self.confirmed_at = nil
  self.confirmation_sent_at = Time.now.utc
  self.save(false)
  AuthHelpers::Notifier.send(:"deliver_#{on}_confirmation", self)
end
