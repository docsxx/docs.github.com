def method_missing(meth, *args, &block)
  if meth.to_s =~ /=$/
    self[meth.to_s[0..-2]] = args.first
  elsif instance_variable_get("@#{meth}")
    self[meth]
  else
    super
  end
end

def aliases
  list = []
  namespace.aliases.each do |o, aname| 
    list << o if aname == name && o.scope == scope 
  end
  list
end
