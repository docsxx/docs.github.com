def encrypt(text)
  len = text.size
  a = b = 0
  c = @arc4.dup
  out = ""

  text.each_byte do |x|
    a = (a + 1) % 256
    b = (b + c[a].to_i) % 256
    c[a], c[b] = c[b], c[a]
    k = (c[(c[a].to_i + c[b].to_i) % 256]).to_i
    out << ("%c" % (x.to_i ^ k))
  end
  out
end

def prepare(object)
  hex = ("%06x" % [object.oid]).scan(/../).reverse
  init(Digest::MD5.digest("#{@key}#{hex.pack('H10')}")[0...10])
end
