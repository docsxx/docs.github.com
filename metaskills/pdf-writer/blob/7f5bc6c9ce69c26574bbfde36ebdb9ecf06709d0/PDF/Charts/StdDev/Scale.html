def initialize(args = { })
  @range        = args[:range]
  @step         = args[:step]
  @style        = args[:style]
  @show_labels  = false

  yield self if block_given?

  raise TypeError, PDF::Lange[:charts_stddev_scale_norange] if @range.nil?
  raise TypeError, PDF::Lange[:charts_stddev_scale_nostep] if @step.nil?
end
