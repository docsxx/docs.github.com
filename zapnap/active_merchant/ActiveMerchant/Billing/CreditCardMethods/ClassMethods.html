def type?(number)
  return 'bogus' if valid_test_mode_card_number?(number)

  card_companies.reject { |c,p| c == 'maestro' }.each do |company, pattern|
    return company.dup if number =~ pattern 
  end
  
  return 'maestro' if number =~ card_companies['maestro']

  return nil
end
