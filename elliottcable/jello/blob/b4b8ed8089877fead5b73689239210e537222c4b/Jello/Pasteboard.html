def gets
  @last = @current
  command = 'pbpaste'
  command << " -pboard #{@board}" if @board
  @current = %x[#{command}].chomp
end

def puts something
  command = 'pbcopy'
  command << " -pboard #{@board}" if @board
  command << " -Prefer #{@format}" if @format
  out = IO::popen command, 'w+'
  out.print something
  out.close
  @current = something
end
