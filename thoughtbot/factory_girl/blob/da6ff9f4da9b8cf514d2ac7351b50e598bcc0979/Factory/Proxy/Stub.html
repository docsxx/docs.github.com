  @stub = klass.new
  @stub.id = next_id
  @stub.instance_eval do
    def new_record?
      id.nil?
    end

    def connection
      raise "stubbed models are not allowed to access the database"
    end

    def reload
      raise "stubbed models are not allowed to access the database"
    end
  end
end
