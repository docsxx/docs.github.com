def self.form_word(*keywords)
  for word in keywords
    word = word.id2name
    module_eval "def \#{word}(options=nil,&block)\ne = @@base.current_element.last\nreturn unless e\nunless e.respond_to? \"\#{word}\"\n@@base.puts \"<!-- FORM ERROR: '\" + e.classname + \"' does not have a '\#{word}' -->\"\nreturn\nend\nreturn e.\#{word}(options,&block) if block_given?\nreturn e.\#{word}(options) if options\nreturn e.\#{word}()\nend\n"
  end
end
