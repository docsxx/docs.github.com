def execute
  return if @files.empty?
  FileUtils.mkdir_p @dest
  create_index(File.join(@dest, "index.html"))

  each_file_pair_sorted do |filename, fileinfo|
    create_file(File.join(@dest, mangle_filename(filename)), fileinfo)
  end
end
