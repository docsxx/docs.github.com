def self.readable_info (info)
  if info.respond_to?(:to_sym)
    info.to_sym.inspect
  else
    info.inspect
  end
end

def resembles? (spec)
  min_length = [spec.length, to_a.length].min
  to_a[0,min_length] == spec[0,min_length]
end
