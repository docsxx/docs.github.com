def add(host, ip = "127.0.0.1", force = false)
  if find_by_host(host).nil? || force
    `#{CreateCmd % [host, ip]}`
    flush!
    find_by_host(host)
  else
    raise "Can not overwrite existing record"
  end      
end

def find_by_host(host)
  @hosts ||= {}
  @hosts[host] ||= begin
    output = `#{ReadCmd % host}`
  
    if output =~ /eDSRecordNotFound/
      nil
    else
      host = parse_host(output)
      ip = parse_ip(output)
    
      Host.new(host, ip)
    end
  end
end
