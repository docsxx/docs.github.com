Module: CodeRay::PluginHost

A simple subclass plugin system. 

 Example:
   class Generators < PluginHost
     plugin_path 'app/generators'
   end

   class Generator
     extend Plugin
     PLUGIN_HOST = Generators
   end

   class FancyGenerator < Generator
     register_for :fancy
   end

   Generators[:fancy]  #-> FancyGenerator
   # or
   CodeRay.require_plugin 'Generators/fancy'

def host_by_id host_id
  unless PLUGIN_HOSTS_BY_ID.default_proc
    ph = Hash.new do |h, a_host_id|
      for host in PLUGIN_HOSTS
        h[host.host_id] = host
      end
      h.fetch a_host_id, nil
    end
    PLUGIN_HOSTS_BY_ID.replace ph
  end
  PLUGIN_HOSTS_BY_ID[host_id]
end

def inspect
  map = plugin_hash.dup
  map.each do |id, plugin|
    map[id] = plugin.to_s[/(?>[\w_]+)$/]
  end
  "#{name}[#{host_id}]#{map.inspect}"
end

def list
  Dir[path_to('*')].select do |file|
    File.basename(file)[/^(?!_)\w+\.rb$/]
  end.map do |file|
    File.basename file, '.rb'
  end
end

def map hash
  for from, to in hash
    from = validate_id from
    to = validate_id to
    plugin_hash[from] = to unless plugin_hash.has_key? from
  end
end

def register plugin, *ids
  for id in ids
    unless id.is_a? Symbol
      raise ArgumentError,
        "id must be a Symbol, but it was a #{id.class}"
    end
    plugin_hash[validate_id(id)] = plugin
  end
end
