def build_flags(*flags)
  _flags = *flags

  unless _flags.is_a?(Integer)
    _flags = MAGIC_NONE

    flags.flatten.each { |flag|
      if value = flag.is_a?(Integer) ? flag : MAGIC_FLAGS[flag.to_sym]
        _flags |= value
      else
        raise ArgumentError, "#{value.nil? ? 'no such flag' : 'flag not available'}: #{flag}"
      end
    }
  end

  _flags
end

def open(*flags)
  fm = new(*flags)

  if block_given?
    begin
      yield fm
    ensure
      fm.close
    end
  end

  fm
end
