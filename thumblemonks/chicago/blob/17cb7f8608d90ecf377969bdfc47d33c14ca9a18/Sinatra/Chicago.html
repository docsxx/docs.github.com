def catch_all_css(path='/stylesheets')
  get("#{path}/*.css") do
    begin
      content_type 'text/css'
      sass params["splat"].first.to_sym
    rescue Errno::ENOENT
      pass
    end
  end
end

def get_obvious(name)
  get "/#{name}" do
    haml name.to_sym
  end
end
