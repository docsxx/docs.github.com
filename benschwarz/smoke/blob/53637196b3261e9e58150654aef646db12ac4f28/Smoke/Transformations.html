def discard(key, matcher)
  @items.reject! {|i| (i[key] =~ matcher) ? true : false }
  return self
end

def sort(key)
  @items = @items.sort_by{|i| i[key] }
rescue NoMethodError => e
  Smoke.log.info "You're trying to sort by \"#{key}\" but it does not exist in your item set"
ensure
  return self
end
