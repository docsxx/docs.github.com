def self.load_class(clazz)
  clazz = clazz.split("::").inject(Kernel) do |container, name|
    container.const_get(name.to_s)
  end
rescue NameError
  raise Sinatra::NotFound
end

def self.query_from_params(params)
  query = nil
  unless (params.nil?)
    query = params.clone
    query.delete('id')
    query.delete('class')
    query.delete('prop')
    query.delete('rel')
  end
  query
end
