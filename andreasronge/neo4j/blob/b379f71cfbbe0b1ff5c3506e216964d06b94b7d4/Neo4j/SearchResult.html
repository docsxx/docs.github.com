Class: Neo4j::SearchResult

Included Modules

Enumerable

Holds the search result and performs the query. The query is not performed until a search result item is requested. It is possible that one node is deleted,updated,created after the search is performed.

def each
    hits.each do |doc|
      id = doc[:id]
      node = Neo4j.instance.find_node(id.to_i)
      # can happen that another thread has deleted it
      raise LuceneIndexOutOfSyncException.new("lucene found node #{id} but it does not exist in neo") if node.nil?
      yield node
    end
end

def query_params
  unless @sort_by_fields.empty?
    return case @query
      when Hash
        @query.merge({:sort_by => @sort_by_fields}) 
      when String
        [@query, { :sort_by => @sort_by_fields}]  
    end
  end
  return @query
end
