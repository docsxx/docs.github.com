def self.parse_type(type_name)
    # FIXME plucked from SQLite driver, this needs to be in DBI proper
    return ['varchar'] unless type_name
    type_name.match(/^([^\(\s]+)\s*(\(\s*(\d+)\s*(,\s*(\d+))?\s*\))?$/)
end
