def aasm_write_state(state)
  old_value = read_attribute(self.class.aasm_column)
  write_attribute(self.class.aasm_column, state.to_s)

  unless self.save(false)
    write_attribute(self.class.aasm_column, old_value)
    return false
  end

  true
end
