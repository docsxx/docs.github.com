def read(len = nil)
  return nil if @read >= @size
  len ||= @size - @read
  max_read = [len, @size - @read].min
  ret = @io.read(max_read)
  @read += ret.size
  ret
end

def rewind
  raise NonSeekableStream unless @io.respond_to?(:pos=)
  @io.pos = @orig_pos
  @read = 0
end
