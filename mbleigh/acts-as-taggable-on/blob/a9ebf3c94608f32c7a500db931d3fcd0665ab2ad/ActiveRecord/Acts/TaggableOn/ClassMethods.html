def acts_as_taggable_on(*args)
  args.flatten! if args
  args.compact! if args
  for tag_type in args
    tag_type = tag_type.to_s
    # use aliased_join_table_name for context condition so that sphix can join multiple
    # tag references from same model without getting an ambiguous column error
    self.class_eval do
      has_many "#{tag_type.singularize}_taggings".to_sym, :as => :taggable, :dependent => :destroy, 
        :include => :tag, :conditions => ['#{aliased_join_table_name rescue "taggings"}.context = ?',tag_type], :class_name => "Tagging"
      has_many "#{tag_type}".to_sym, :through => "#{tag_type.singularize}_taggings".to_sym, :source => :tag
    end
    
    self.class_eval "def self.taggable?\ntrue\nend\n\ndef self.caching_\#{tag_type.singularize}_list?\ncaching_tag_list_on?(\"\#{tag_type}\")\nend\n\ndef self.\#{tag_type.singularize}_counts(options={})\ntag_counts_on('\#{tag_type}',options)\nend\n\ndef \#{tag_type.singularize}_list\ntag_list_on('\#{tag_type}')\nend\n\ndef \#{tag_type.singularize}_list=(new_tags)\nset_tag_list_on('\#{tag_type}',new_tags)\nend\n\ndef \#{tag_type.singularize}_counts(options = {})\ntag_counts_on('\#{tag_type}',options)\nend\n\ndef \#{tag_type}_from(owner)\ntag_list_on('\#{tag_type}', owner)\nend\n\ndef find_related_\#{tag_type}(options = {})\nrelated_tags_for('\#{tag_type}', self.class, options)\nend\nalias_method :find_related_on_\#{tag_type}, :find_related_\#{tag_type}\n\ndef find_related_\#{tag_type}_for(klass, options = {})\nrelated_tags_for('\#{tag_type}', klass, options)\nend\n\ndef top_\#{tag_type}(limit = 10)\ntag_counts_on('\#{tag_type}', :order => 'count desc', :limit => limit.to_i)\nend\n"
  end      
  
  if respond_to?(:tag_types)
    write_inheritable_attribute( :tag_types, (tag_types + args).uniq )
  else
    self.class_eval do
      write_inheritable_attribute(:tag_types, args.uniq)
      class_inheritable_reader :tag_types
    
      has_many :taggings, :as => :taggable, :dependent => :destroy, :include => :tag
      has_many :base_tags, :class_name => "Tag", :through => :taggings, :source => :tag
    
      attr_writer :custom_contexts
    
      before_save :save_cached_tag_list
      after_save :save_tags
      
      if respond_to?(:named_scope)
        named_scope :tagged_with, lambda{ |tags, options|
          find_options_for_find_tagged_with(tags, options)
        }
      end
    end
    
    include ActiveRecord::Acts::TaggableOn::InstanceMethods
    extend ActiveRecord::Acts::TaggableOn::SingletonMethods                
    alias_method_chain :reload, :tag_list
  end
end
