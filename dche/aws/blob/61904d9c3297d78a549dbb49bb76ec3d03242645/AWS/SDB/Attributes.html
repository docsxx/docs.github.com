def attr(name)
  raise RuntimeError, 'Item name is not specified.' if @item.nil?
  
  unless name.nil? || name.empty?
    hash = @o[@item]
    hash[name] = [] if hash[name].nil?
    @attr = name
  end
  self
end

def to_queries(item = nil)
  ic = 0 # counter for item
  @o.each_key.inject({}) do |q, item_name|
    ac = 0 # counter for attribute
    if item  && item_name == item
      @o[item_name].each do |n, v|
        case v
        when Array
          v.each do |val| 
            q[attribute_query(:name, ac)] = n
            q[attribute_query(:value, ac)] = val
            ac += 1
          end
        else
          q[attribute_query(:name, ac)] = n
          q[attribute_query(:value, ac)] = v
          q[attribute_query(:replace, ac)] = 'true'
        end
      end
    elsif item.nil? && !@o[item_name].empty?
      q[item_name_query(ic)] = item_name
      @o[item_name].each do |n, v|
        case v
        when Array
          v.each do |val| 
            q[attribute_query(:name, ac, ic)] = n
            q[attribute_query(:value, ac, ic)] = val
            ac += 1
          end
        else
          q[attribute_query(:name, ac, ic)] = n
          q[attribute_query(:value, ac, ic)] = v
          q[attribute_query(:replace, ac, ic)] = 'true'
        end
      end
      ic +=1
    end
    q
  end
end

def val(value)
  raise RuntimeError, 'Attribute is not specified.' if @attr.nil?
  
  vals = @o[@item][@attr]
  if Array === vals
    vals << value
  else
    @o[@item][@attr] = [vals, value]
  end
  self
end
