def initialize(output, sdb = nil)
  @output = nil
  
  case output
  when Array
    @output = output.flatten.map {|val| normalized_name(val)}.join(",")
  when :all
    @output = '*'
  when :count
    @output = 'count(*)'
  when :item_name
    @output = 'ItemName()'
  end
  raise ArgumentError, "Invalid output list type." if @output.nil?
  
  @operators = {}
  @stack = []
  @sort = @limit = ''
  @sdb = sdb
  @domain = sdb.nil? ? '' : sdb.default_domain
end

def to_s
  exp = @stack.last && !@stack.last.empty? ? "where #{@stack.last}" : ''
  "select #{@output} from #{@domain} #{exp} #{@sort} #{@limit}".rstrip
end
