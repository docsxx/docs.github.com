AWS::QueryRequestMixin

A simple DSL for accessing Amazon SimpleDB service.

Examples: 

 sdb = SDB.new

#. List domains 

 sdb.domains.each {|name| puts name}

#. Create a domain 

 sdb.create_domain "512wawa"

#. Delete a domain 

 sdb.delete_domain "cpc"

#. Inspect the metadata of a domain 

 puts sdb.domain_meta("512wawa")['ItemCount']

#. Set the default domain used in subsequent requests. 

 sdb.use_domain("512wawa")

#. Get all attributes of item. 

 sdb.item("myChild")

#. Add attributes to an item. 

 sdb.put()

def domain_metadata(name)
  req('Action' => 'DomainMetadata', 'DomainName' => name) do |doc|
    (doc%'DomainMetadataResult').children.inject({}) do |md, e|
      if Hpricot::Elem === e
        md[e.name] = e.inner_text
      end
      md
    end
  end

def domains(opts = {})
  next_token = nil
  names = []
  
  req({'Action' => 'ListDomains'}.merge(opts)) do |doc|
    names = (doc/'DomainName').map {|e| e.inner_text}
    elm = (doc%'NextToken')
    next_token = elm && elm.inner_text
  end
  
  next_token.nil? ? names : names + domains('NextToken' => next_token)       
end

def execute_select(expr, opts = {})
  res = Attributes.new
  queries = {'Action' => 'Select', 'SelectExpression' => expr.to_s}
  queries.merge!({'NextToken' => opts[:NextToken].to_s}) if opts[:NextToken]
  
  new_nt = nil
  xml = req(queries, :raw => opts[:raw])
  
  return xml if opts[:raw]
  
  (xml/'Item').each do |item|
    res.item((item%'Name').inner_text)
    (item/'Attribute').each do |attr|
      res.attr((attr%'Name').inner_text)
      res.val((attr%'Value').inner_text)
    end
  end
  new_nt = (xml%'NextToken') && (xml%'NextToken').inner_text
  
  new_nt.nil? ? res.to_hash : res.to_hash.merge!(select(expr, {:NextToken => new_nt}))
end

def get(item, attributes)
  queries = {'Action' => 'GetAttributes', 'DomainName' => default_domain, 'ItemName' => item}
  req(queries.merge!(attr_name_queries(attributes))) do |doc|
    (doc/'GetAttributesResult/Attribute').inject({}) do |memo, elm|
      name = (elm%'Name').inner_text
      value = (elm%'Value').inner_text
      
      ov = memo[name]
      if Array === ov
        ov << value
      elsif ov
        memo[name] = [ov, value]
      else
        memo[name] = value
      end
      memo
    end         
  end
end

def put(attributes, item_name = nil)
  queries = {'DomainName' => default_domain}
  if item_name
    queries.merge!({'Action' => 'PutAttributes', 'ItemName' => item_name})
  else
    queries.merge!({'Action' => 'BatchPutAttributes'})
  end
  
  req queries.merge!(attr_queries(item_name, attributes))
  self
end
