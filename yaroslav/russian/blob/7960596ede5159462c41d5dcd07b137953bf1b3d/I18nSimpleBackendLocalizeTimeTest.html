def setup
  @old_timezone, ENV['TZ'] = ENV['TZ'], 'UTC'
  @backend = I18n::Backend::Simple.new
  add_datetime_translations
  @morning = Time.parse '2008-01-01 6:00 UTC'
  @evening = Time.parse '2008-01-01 18:00 UTC'
end
