def initialize( resolution )
  @resolution = resolution
  @queue = []

  @thread = Thread.new do
    loop do
      dispatch
      sleep @resolution
    end
  end
end

def at( time, &block )
  time = time.to_f if time.kind_of? Time
  @queue.push [ time, block ]
end
