def find_in_with_js(dir)
  results = find_in_without_js(dir)

  Dir.glob("#{dir}/*") do |item|
    case
    when File.basename(item)[0] == ?.
    when File.directory?(item)
    when item =~ /\.js$/
      results.update(extract_annotations_from(item, /(?:\/\/|\/\*)\s*(#{tag})(?:.*:|:)?\s*(.*?)(?:\*\/)?$/))
    end
  end

  results
end
