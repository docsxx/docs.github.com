Module: Ramaze::Logging

This module provides a basic skeleton for your own loggers to be compatible. The minimal usage is like this: 

  class MyLogger
    include Logging

    def log(tag, *args)
      p tag => args
    end
  end

def error(ex)
  if ex.respond_to?(:exception)
    message = ex.backtrace
    message.map!{|m| m.to_s.gsub(/^#{Regexp.escape(Dir.pwd)}/, '.') }
    message.unshift(ex.inspect)
  else
    message = ex.to_s
  end
  tag_log(:error, :to_s, *message)
end

def tag_log(tag, meth, *msgs)
  msgs.each do |msg|
    string = (msg.respond_to?(:to_str) ? msg : msg.send(meth))
    log(tag, string)
  end
end
