def call(cooldown)
  if cooldown and Time.now > @last + cooldown
    yield
    @last = Time.now
  end
end

def changed_files
  @files.each do |file, stat|
    if new_stat = safe_stat(file)
      if new_stat.mtime > stat.mtime
        @files[file] = new_stat
        yield(file)
      end
    end
  end
end

def watch(file)
  return true if watching?(file) # if already watching
  if stat = safe_stat(file)
    @files[file] = stat
  end
end
