def initialize(*loggers)
  @loggers = loggers
  @ignored_tags = Set.new
  @loggers.map! do |logger|
    next(nil) if logger == self
    logger.is_a?(Class) ? logger.new : logger
  end
  @loggers.uniq!
  @loggers.compact!
end

def log(tag, *args)
  return if @ignored_tags.include?(tag)
  @loggers.each do |logger|
    logger.log(tag, *args)
  end
end
