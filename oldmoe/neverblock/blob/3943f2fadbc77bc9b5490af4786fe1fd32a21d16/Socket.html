def connect_neverblock(server_sockaddr)
  begin
    connect_nonblock(server_sockaddr)
  rescue Errno::EINPROGRESS, Errno::EINTR, Errno::EALREADY, Errno::EWOULDBLOCK
    NB.wait(:write, self)
    retry
  rescue Errno::EISCONN
    # do nothing, we are good

  end
end
