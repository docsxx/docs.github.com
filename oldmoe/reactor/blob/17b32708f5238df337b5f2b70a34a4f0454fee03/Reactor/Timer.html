def initialize(timers, time, periodical, &block)
  @timers = timers
  @time = time * 1000
  @periodical = periodical
  @block = block
  @active = true
  add_to_timers
end

def add_to_timers
  @time_of_fire = (Time.now.to_f * 1000).to_i + @time
  @timers.insert_sorted(self)      
end
