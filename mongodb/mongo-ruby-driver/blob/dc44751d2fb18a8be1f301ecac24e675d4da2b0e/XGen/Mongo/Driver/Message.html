def initialize(op)
  @op = op
  @message_length = HEADER_SIZE
  @data_length = 0
  @request_id = (@@class_req_id += 1)
  @response_id = 0
  @buf = ByteBuffer.new
  
  @buf.put_int(16)      # holder for length
  @buf.put_int(@request_id)
  @buf.put_int(0)       # response_to
  @buf.put_int(op)
end

def update_message_length
  pos = @buf.position
  @buf.put_int(@buf.size, 0)
  @buf.position = pos
end
