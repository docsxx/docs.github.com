def initialize(name, type = nil, category = nil)
  super :identity

  if name.is_a?(XML::Node)
    self.inherit name
  else
    self.name = name
    self.type = type
    self.category = category
  end
end

def eql?(other)
  other.kind_of?(self.class) &&
  other.name == self.name &&
  other.type == self.type &&
  other.category == self.category
end
