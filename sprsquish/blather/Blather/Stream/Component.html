def receive(node) # :nodoc:
  if node.element_name == 'handshake'
    @client.stream_started(self)
  else
    super
  end

  if node.element_name == 'stream:stream'
    send("<handshake>#{Digest::SHA1.hexdigest(@node['id']+@pass)}</handshake>")
  end
end
