def method_missing(method, opt = ;{}, &block)
  case method.to_s
  when /^create_(.+)$/
    create_named($1.to_sym, opt, &block)
  when /^new_(.+)$/
    new_named($1.to_sym, opt, &block)
  else
    raise NameError, "no such method `#{method}'"
  end
end
