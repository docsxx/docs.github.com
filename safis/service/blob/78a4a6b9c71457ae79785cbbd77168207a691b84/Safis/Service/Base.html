def start!
  log.info 'Starting service.'
  
  # Handle SIGINT as a means of gracefully stopping the app
  Signal.trap('INT') do
    # spawn this in another thread so we don't block the main thread from finishing up
    Thread.new do
      self.stop!
    end
  end
  
  # Handle SIGTERM as a means of gracefully stopping the app
  Signal.trap('TERM') do
    # spawn this in another thread so we don't block the main thread from finishing up
    Thread.new do
      self.stop!
    end
  end
  
  @running = true
  @delegate.service_did_start
  @running = false
  
  # prod the shutdown thread, if need be
  @shutdown_thread.run unless @shutdown_thread.nil?
  log.info 'Service gracefully stopped.'
end

def stop!
    log.info 'Stopping service.'
    
    # Start our timer
    shutdown_timeout = @config[:service, :shutdown_timeout] || 30
    shutdown_before  = Time.now + shutdown_timeout
    
    # Spawn a thread to kill us off if we go over
    @shutdown_thread = Thread.new do
      while ( @running and Time.now < shutdown_before ) do
        sleep_time = shutdown_before - Time.now
        sleep(sleep_time)
      end
      
      # No good, we need to terminate
      if ( @running ) then
        log.fatal "Couldn't gracefully shut down the service within #{shutdown_timeout} seconds.  Terminating."
        exit
      end
    end
    
    @delegate.service_will_stop
    
    @shutdown_thread.join
  end
end
