Class: Safis::Service::Config

Constructor

public initialize(search_paths, mode)

Loads in a configuration, running over a set of search paths for YAML configuration files.

This class checks each path for a ‘default.yaml’, as well as ’<service mode>.yaml’ (for example, debug.yaml), in each search path.

All default.yaml configurations are first read in and merged, followed by mode-specific configurations. Later configurations override earlier ones.

def initialize(search_paths, mode)
  # Build a list of config files to read in
  config_files = []
  # Two passes.  default.yaml first
  search_paths.each do |path|
    default_path = File.join(path, 'default.yaml')
    if ( File.exist? default_path ) then
      config_files << default_path
    end
  end
  
  # <mode>.yaml next
  search_paths.each do |path|
    mode_path = File.join(path, "#{mode}.yaml")
    if ( File.exist? mode_path ) then
      config_files << mode_path
    end
  end
  
  @config = {}
  # Now iterate over them and merge them into our config keys
  config_files.each do |file|
    log.info "Reading configuration: #{file}"
    
    begin
      single_config = YAML.load_file(file)
      
    rescue ArgumentError => err
      log.fatal "Unable to parse configuration: #{file}", err
      raise err
    end
    
    if ( not single_config.kind_of? Hash ) then
      message = "Expected a Hash for the root of #{file}."
      log.fatal "#{message}  Instead, we got:", single_config
      raise MergeError, message
    end
    
    # Merge this configuration into the overall configuration
    self.merge(single_config)
  end
end

def [](*config_path)
  position = @config
  config_path.each do |key|
    return nil unless position.respond_to? :has_key?
    
    position = position[key]
  end
  
  return position
end

def merge(hash, *config_path)
  node = self[*config_path]
  path = config_path.join('/')
  
  # This shouldn't happen...
  if ( node.nil? ) then
    message = "Attempted to merge into a non-existant key at #{path}!"
    log.fatal message
    raise MergeError, message
  end
  
  hash.each do |key, value|
    # ensure that keys are symbols
    key      = key.to_sym
    key_path = "#{path}/#{key}"
    
    existing = node[key]
    
    # do nothing if the value is nil
    if ( not value.nil? ) then
      # should we merge?
      if ( existing.kind_of? Hash or existing.kind_of? Array ) then
        # ensure they're the same time
        if ( existing.class != value.class ) then
          message = "Cannot merge a #{value.class} with a #{existing.class} at #{key_path}"
          log.fatal message
          raise MergeError, message
        end
        
        # do the merge for an Array?
        if ( existing.kind_of? Array ) then
          node[key] = existing.concat(value)
        
        # Nope, recursive Hash merge
        else
          self.merge(value, *[*config_path, key])
        end
      
      # if this is a hash, convert keys to symbols by recursively calling us
      elsif ( value.kind_of? Hash ) then
        node[key] = {}
        self.merge(value, *[*config_path, key])
      
      # otherwise, replace
      else
        node[key] = value
      end
    end
  end
end
