Class: Safis::Util::Requester

A Requester is a simple class that wraps the Curl::Multi interface of the curb gem.

It runs in a separate thread, providing callbacks to interesting events and allowing you to both run synchronous and asynchronous web requests.

The maximum number of connections that can be concurrently processed. Please note that cURL may cache these connections by trying to keep the socket open as long as possible. (e.g. CURLMOPT_MAXCONNECTS is also set)

def initialize(max_connections = 25)
        # connection tracking
        @max_connections = max_connections
        
        @handle_mutex    = Mutex.new
        @pending_handles = []
        
        # cURL setup
        @multi = Curl::Multi.new
        @multi.max_connects = @max_connections
        
        # Set up our perform thread
        @perform_thread = Thread.new do
          while (true) do
            # sleep unless there's something to do
            should_sleep = false
            @handle_mutex.synchronize do
              should_sleep = true if @multi.idle?
            end
            sleep if should_sleep
            
            # and then do some work.  Note that this call blocks until ALL requests are handled!
            @multi.perform
          end
        end
      end
      
      # Asynchronously processes a request.
      # 
      # Your existing cURL event callbacks WILL be honored.
      # 
      # @param [Curl::Easy] request  The request to process.
      def process_async(request)
        # set up the callbacks
        old_on_complete = request.on_complete do |request|
          # exceptions within this thread indefinitely block and turn us into a black hole..
          begin
            log.debug "Request on_complete: #{request.url}"
            
            # We need to enqueue outside of the critical section so we don't deadlock.
            new_request = nil
            @handle_mutex.synchronize do
              if ( @pending_handles.length > 0 ) then
                new_request = @pending_handles.shift
              end
            end
            self.enqueue(new_request) unless new_request.nil?
            
            # make sure the perform thread is checking
            @perform_thread.run
            
            # Give the old callback handle a chance to go
            begin
              old_on_complete.call(request) unless old_on_complete.nil?
            
            rescue Exception => err
              log.error 'Encountered an error in request.on_complete:', err
              
              # Have the wait thread throw this, if there is one.  Otherwise the exception gets
              # eaten, unfortunately.
              @handle_mutex.synchronize do
                @exception = err
              end
              
              @blocking_thread.run unless @blocking_thread.nil?
            end
            
            # prod the blocking thread, if there is one
            if ( not @blocking_thread.nil? and @multi.idle? ) then
              @blocking_thread.run
            end
            
          # this is our 'oh shit!' handler.  
          rescue Exception => err
            # Unfortunately, if we throw an exception here, it deadlocks this thread...
            log.error 'Ran into an exception while processing a request!', err
          end
        end
        
        should_enqueue = false
        @handle_mutex.synchronize do
          # do we need to queue?
          if ( @multi.requests.length >= @max_connections ) then
            @pending_handles << request
            
          # nope. start it up
          else
            should_enqueue = true
          end
        end
        
        if ( should_enqueue ) then
          self.enqueue(request)
          # and prod the perform thread
          @perform_thread.run
        end
      end
      
      # Synchronously processes a request.
      # 
      # Your existing cURL event callbacks WILL be honored.
      # 
      # @param [Curl::Easy] request  The request to process.
      def process_sync(request)
        done = false
        
        wait_thread = Thread.new do
          while ( not done ) do
            sleep
          end
        end
        
        old_on_complete = request.on_complete do |request|
          # unstick our blocking thread
          done = true
          wait_thread.run
          
          old_on_complete.call(request) unless old_on_complete.nil?
        end
        
        self.process_async(request)
        
        wait_thread.join
      end
      
      # Blocks until the requester has finished processing its queue
      def block_until_done
        @blocking_thread = Thread.current
        
        catch :breakout do
          while ( true ) do
            escape = false
            @handle_mutex.synchronize do
              # if we're done, or have an error, we break out of this loop
              if ( @multi.idle? or not @exception.nil? ) then
                escape = true
              end
            end
            
            throw :breakout if escape
            
            sleep
          end
        end
        
        @handle_mutex.synchronize do
          # did we have an error?
          if ( not @exception.nil? ) then
            # clear out the multi handle
            @multi.cancel!
            
            # and throw our exception
            raise @exception
          end
        end
        
        @blocking_thread = nil
      end
      
    protected
      # enqueues a request with curl and the active handles array.
      # 
      # Assumes that this is in a critical section.
      def enqueue(request)
        log.debug "Request enqueued: #{request.url}"
        @multi.add(request)
      end
    end
  end
end
