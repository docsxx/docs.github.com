def drain!
  # short circuit if we don't have anything to do
  return nil if @tasks.length == 0
  
  log.debug "Run loop drain!"
  
  to_delete = []
  
  earliest_next_run = nil
  
  @tasks.each do |task|
    next_run = task.next_run?
    
    # should we run this?
    if ( next_run.nil? ) then
      task.run!
      to_delete << task
    
    # otherwise, we need to figure out how long to sleep
    else
      earliest_next_run ||= next_run
      
      if ( next_run < earliest_next_run ) then
        earliest_next_run = next_run
      end
    end
  end
  
  to_delete.each do |task|
    @tasks.delete(task)
  end
  
  log.debug 'Run loop finished draining.'
  
  return earliest_next_run
end

def run!
  log.info 'Run loop starting.'
  
  @stop = false
  
  @main_thread = Thread.new do
    log.debug 'Main thread start'
    @running = true
    
    loop do
      break if not @stop_thread.nil?
      
      next_run = self.drain!
      
      break if not @stop_thread.nil?
      
      sleep_time = MAXIMUM_SLEEP
      if ( not next_run.nil? ) then
        # Wake up for the next run, if there is one within our sleep duration
        next_run_offset = [next_run - Time.now, 0].max
        sleep_time      = [MAXIMUM_SLEEP, next_run_offset].min
      end
      
      # wait until we have something to do
      log.debug "Sleeping for #{sleep_time} seconds."
      sleep(sleep_time)
    end
    
    @running = false
    log.debug 'Main thread stopped.'
    
    @stop_thread.run unless @stop_thread.nil?
  end
  
  @main_thread.join
  log.info 'main thread returned'
  @main_thread = nil # We probably won't get here, but let's play it safe.
end

public stop!(timeout = nil, &callback)

Attempts to gracefully shut down the run loop within the specified amount of time.

By default, this will block until the run loop completes. However, if you call stop! from the run loop, this call won’t block.

The block that’s passed in will be called once the run loop stops or is terminated.

def stop!(timeout = nil, &callback)
  shutdown_work = Proc.new do
    log.info 'Stopping run loop.'
    
    stop_before = (timeout.nil?) ? nil : Time.now + timeout
    
    # build a thread that we can use to wait for the main thread to stop
    thread = Thread.new do
      while ( @running and (stop_before.nil? or Time.now < stop_before) )
        sleep_time = (stop_before.nil?) ? MAXIMUM_SLEEP :
                                          [MAXIMUM_SLEEP, stop_before - Time.now].min
        
        sleep(sleep_time)
      end
      
      # kill the run loop thread if it goes past
      if ( @running ) then
        log.fatal "Run loop didn't stop within #{timeout} seconds.  Terminating run loop."
        @main_thread.terminate
      end
    end
    
    @stop_thread = thread
    self.prod!
    
    @stop_thread.join
    @stop_thread = nil
    
    callback.call if not callback.nil?
  end
  
  if ( Thread.current == @main_thread ) then
    Thread.new do
      shutdown_work.call
    end
  else
    shutdown_work.call
  end
end
