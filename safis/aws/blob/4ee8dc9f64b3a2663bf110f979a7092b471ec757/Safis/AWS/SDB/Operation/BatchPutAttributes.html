public append(item_attributes = {})

Appends/replaces attributes for items.

This takes a relatively complex hash of values, that should look like: 

  <item_name> => {
    <attribute> => {
      :values => [...],
      :replace => true|false,
    },
    ...
  }

The :replace attribute is optional (defaults to false). If you specify conflicting values of replace for a given item/attribute pair, it will fall back to merging the new attributes in.

def append(item_attributes = {})
    @pending_items ||= {}
    
    item_attributes.each do |item_name, attributes|
      item = (@pending_items[item_name] ||= {})
      
      attributes.each do |attribute_name, info|
        attribute = (@pending_items[item_name][attribute_name] ||= {:values => []})
        
        attribute[:values].concat(info[:values])
        
        # deal with the replace parameter
        if ( info[:replace] and attribute[:replace] != false ) then
          attribute[:replace] = true
        
        elsif ( not info[:replace] and attribute[:replace] ) then
          attribute[:replace] = false
        end
      end
    end
    
    # spin off a request if we're over our batch size
    while ( @pending_items.length > BATCH_SIZE ) do
      self.process_request
    end
  end
  
protected
  # Returns the request parameters
  # 
  # This will be repeatedly called until nil is returned - allowing for batches.
  def build_request
    return nil unless @pending_items.length > 0
    
    request = super
    
    # Get our base request
    request = super
    request.params['DomainName'] = @domain_name
    
    item_count = 0
    @pending_items.each do |item_name, attributes|
      attribute_count = 0
      
      # the item
      item_base = "Item.#{item_count}"
      request.params["#{item_base}.ItemName"] = item_name
      
      # each attribute
      attributes.each do |attribute, info|
        # replace, if we need to.  Note that this is global to the attribute, apparently.
        if ( info[:replace] ) then
          request.params["#{item_base}.Attribute.#{attribute_count}.Replace"] = 'true'
        end
        
        # each value...
        info[:values].each do |value|
          attr_base = "#{item_base}.Attribute.#{attribute_count}"
          request.params["#{attr_base}.Name"]  = attribute
          request.params["#{attr_base}.Value"] = value
          
          attribute_count += 1
        end
      end
      
      @pending_items.delete(item_name)
      
      item_count += 1
      break if item_count >= BATCH_SIZE
    end
    
    return request
  end
