def receive(*args, &block)
  response = self.queue_perform(Operation::ReceiveMessage, args, block) do |num_messages = 1|
    [num_messages]
  end
  
  messages = []
  response.body.each do |message_info|
    # requester, signer, account_id, queue_name, message_id, receipt_handle, message_body
    messages << Message.new(
      @requester, 
      @signer, 
      @account_id, 
      @queue_name, 
      message_info[:message_id],
      message_info[:receipt_handle],
      message_info[:body]
    )
  end
  
  response.body = messages
  
  return response
end
