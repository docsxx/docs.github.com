def aws_encode(token)
  utf8_token = token.encode('UTF-8')
  
  encoded_token = ''
  utf8_token.chars do |char|
    if ( VALID_TOKEN_CHARS.match(char) ) then
      encoded_token << char
      
    # otherwise encode it
    else
      # hex each byte
      char.bytes.to_a.each do |byte|
        hex = byte.to_s(16)
        hex = '0' + hex if hex.length == 1
        
        encoded_token << "%#{hex.upcase}"
      end
    end
  end
  
  return encoded_token
end

def encode!
  @url_path = @url_path.encode('UTF-8')
  
  # headers
  new_headers = {}
  @headers.each do |key, value|
    new_headers[key.encode('UTF-8')] = value.encode('UTF-8')
  end
  @headers = new_headers
  
  # params
  new_params = {}
  @params.each do |key, value|
    new_params[self.aws_encode(key)] = self.aws_encode(value)
  end
  @params = new_params
end
