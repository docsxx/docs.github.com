def append(num_messages = 1)
    @num_messages ||= 0
    @num_messages += num_messages
    
    while ( true ) do
      break if self.process_request.nil?
    end
  end
  
protected
  # Builds a ReceiveMessage request
  def build_request
    return nil if @num_messages == 0
    
    num_to_request = (@num_messages > MAX_BATCH) ? MAX_BATCH : @num_messages
    
    request = super
    request.params['MaxNumberOfMessages'] = num_to_request.to_s
    
    @num_messages -= num_to_request
    
    return request
  end
  
  # Appends response info onto the body
  def append_response(response_body, response_doc, request)
    response_body ||= []
    
    response_doc.css('Message').each do |message|
      response_body << {
        :message_id     => message.css('MessageId').text,
        :receipt_handle => message.css('ReceiptHandle').text,
        :md5_of_body    => message.css('MD5OfBody').text,
        :body           => message.css('Body').text,
      }
    end
    
    return response_body
  end
end
