def helper_method(*meths)
  meths.flatten.each do |meth|
    master_helper_module.class_eval "def \#{meth}(*args, &blk)\ncontroller.send(%(\#{meth}), *args, &blk)\nend\n", __FILE__, __LINE__ + 1
  end
end
