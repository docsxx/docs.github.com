def initialize(path, options = {})
  raise ArgumentError, "path already is a Path class" if path.is_a?(Path)        
  
  super(options)
  @path, @paths = path, {}
  
  # **/*/** is a hax for symlinked directories
  load_templates("#{@path}/{**/*,**}/**") if @cache
end
