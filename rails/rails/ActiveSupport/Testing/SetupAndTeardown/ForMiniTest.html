def run(runner)
  result = '.'
  begin
    run_callbacks :setup
    result = super
  rescue Exception => e
    result = runner.puke(self.class, self.name, e)
  ensure
    begin
      run_callbacks :teardown, :enumerator => :reverse_each
    rescue Exception => e
      result = runner.puke(self.class, self.name, e)
    end
  end
  result
end
