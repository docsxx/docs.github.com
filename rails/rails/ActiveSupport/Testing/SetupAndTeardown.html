def self.included(base)
  base.class_eval do
    include ActiveSupport::Callbacks
    define_callbacks :setup, :teardown

    if defined?(MiniTest::Assertions) && TestCase < MiniTest::Assertions
      include ForMiniTest
    elsif defined? Spec
      include ForRspec
    else
      include ForClassicTestUnit
    end
  end
end
