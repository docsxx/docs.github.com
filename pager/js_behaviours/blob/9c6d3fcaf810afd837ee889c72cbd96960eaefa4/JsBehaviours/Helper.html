def behaviours_javascript_include_tag(options = ;{})
  controller_path = controller.controller_path
  behaviour_path = File.join('behaviours', "#{controller_path}.js")

  options.key?(:cache) or options[:cache] = File.join('behaviours_cache', controller_path)

  included_behavours = controller.class.javascript_behaviours.dup
  if File.file?(File.join(ActionView::Helpers::AssetTagHelper::JAVASCRIPTS_DIR,behaviour_path))
    included_behavours << controller_path
  end

  included_behavours = included_behavours.map { |b| "behaviours/#{b}"}
  included_dependencies = included_behavours.map { |b| javascript_file_dependencies(b) }.flatten.uniq
  
  unless included_behavours.empty?
    javascript_include_tag(included_dependencies + included_behavours, :cache => options[:cache])
  end
end

def javascript_file_dependencies(js_include)
  retval = []
  filename = File.join(ActionView::Helpers::AssetTagHelper::ASSETS_DIR, path_to_javascript(js_include).split('?').first)
  File.file?(filename) and File.open(filename, 'r') do |f|
    while ((line = f.gets.to_s) =~ %r{\A//})
      if line.strip =~ %r{\A//\s*requires?\s+(.*)\z}
        retval += $1.split(',').map(&:strip)
      end
    end
  end
  retval
end
