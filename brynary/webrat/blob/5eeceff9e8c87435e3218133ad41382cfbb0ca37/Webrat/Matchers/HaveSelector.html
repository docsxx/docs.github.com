def tag_inspect
  options = @options.dup
  count = options.delete(:count)
  content = options.delete(:content)

  html = "<#{@expected}"
  options.each do |k,v|
    html << " #{k}='#{v}'"
  end

  if content
    html << ">#{content}</#{@expected}>"
  else
    html << "/>"
  end

  html
end
