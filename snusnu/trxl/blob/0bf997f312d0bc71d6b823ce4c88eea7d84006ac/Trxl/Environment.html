def [](variable)

    if env.has_key?(var)
      return env[var]
    end
  end
  nil

def []=(variable, value)
  var = variable.to_sym
  # search all scopes
  @stack.each do |env|
    if env.has_key?(var)
      return env[var] = value 
    end
  end
  # not found, assign it in local scope
  local[var] = value
end
