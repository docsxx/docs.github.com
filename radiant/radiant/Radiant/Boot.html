def load_initializer
  begin
    require 'radiant'
    require 'radiant/initializer'
  rescue LoadError => e
    $stderr.puts %(Radiant could not be initialized. #{load_error_message})
    exit 1
  end
  Radiant::Initializer.run(:set_load_path)
end
