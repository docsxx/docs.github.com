def initialize(args=[])
  raise ArgumentError, "You must specify an extension to install." if args.blank?
  self.extension_name = to_extension_name(args.shift)
  if installed?
    puts "#{extension_name} is already installed."
  else
    find_extension && extension.install
  end
end
