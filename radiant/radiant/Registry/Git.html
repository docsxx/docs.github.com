def checkout
  if project_in_git?
    system "git submodule add #{url} vendor/extensions/#{name}"
    system "cd vendor/extensions/#{name}; git submodule init && git submodule update"
  else
    super
    system "cd #{path}; git submodule init && git submodule update"
  end
end
