def only_allow_access_to(*args)
  options = {}
  options = args.pop.dup if args.last.kind_of?(Hash)
  options.symbolize_keys!
  actions = args.map { |a| a.to_s.intern }
  actions.each do |action|
    controller_permissions[action] = options
  end
end
