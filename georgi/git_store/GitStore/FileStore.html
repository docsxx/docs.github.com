Class: GitStore::FileStore

  • GitStore
    • GitStore::FileStore

FileStore reads a working copy out of a directory. Changes made to the store will not be written to a repository. This is useful, if you want to read a filesystem without having a git repository.

def each_blob_in(tree, &blob)
  tree.table.each do |name, entry|
    case entry
    when Blob; yield entry
    when Tree; each_blob_in(entry, &blob)
    end
  end
end

  end

def refresh!
  each_blob_in(root) do |blob|
    path = "#{self.path}/#{blob.path}"
    if File.exist?(path)
      mtime = File.mtime(path)
      if @mtime[blob.path] != mtime
        @mtime[blob.path] = mtime
        blob.load_from_disk
      end
    else
      delete blob.path
    end
  end
end
