def load_from_disk
  @object = nil
  @data = open("#{store.path}/#{path}", 'rb') { |f| f.read }
end

def object=(value)
  @modified = true
  @object = value
  @data = handler.respond_to?(:write) ? handler.write(path, value) : value
end

def write_to_store      
  if modified?
    @modified = false
    @id = store.put_object(data, 'blob')
  else
    @id
  end
end
