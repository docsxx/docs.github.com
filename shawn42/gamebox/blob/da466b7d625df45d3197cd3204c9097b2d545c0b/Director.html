def add_actor(actor)
  @actors << actor
  actor.when :remove_me do
    remove_actor actor
  end
  actor_added actor
  actor
end

def update(time)
  for act in @dead_actors
    @actors.delete act
    actor_removed act
  end
  @dead_actors = []
  for act in @actors
    act.update time if act.alive?
  end
end
