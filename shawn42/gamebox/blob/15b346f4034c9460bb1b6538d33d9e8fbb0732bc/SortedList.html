def add(item)
  added = false
  @list.each_element do |node|
    if node.obj > item
      added = true
      @list.place item, :before, node
      break
    end
  end
  @list << item unless added
  item
end

def contains?(item)

    if obj == item
      return true
    end
  end
  false
end
