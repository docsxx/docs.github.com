def write(dest)
  FileUtils.mkdir_p(File.join(dest, @dir))

  name = @name
  if self.ext != ""
    name = @name.split(".")[0..-2].join('.') + self.ext
  end

  path = File.join(dest, @dir, name)
  File.open(path, 'w') do |f|
    f.write(self.output)
  end
end
