def self.minimize_content(content)

  js_minimizer.output = StringIO.new
  
  js_minimizer.jsmin

end

def action(a)
    if(a==1)
        @output.write @theA
    end
    if(a==1 || a==2)
        @theA = @theB
        if (@theA == "\'" || @theA == "\"")
            while (true)
                @output.write @theA
                @theA = get
                break if (@theA == @theB)
                raise "Unterminated string literal" if (@theA <= "\n")
                if (@theA == "\\")
                    @output.write @theA
                    @theA = get
                end
            end
        end
    end
    if(a==1 || a==2 || a==3)
        @theB = mynext
        if (@theB == "/" && (@theA == "(" || @theA == "," || @theA == "=" ||
                             @theA == ":" || @theA == "[" || @theA == "!" ||
                             @theA == "&" || @theA == "|" || @theA == "?" ||
                             @theA == "{" || @theA == "}" || @theA == ";" ||
                             @theA == "\n"))
            @output.write @theA
            @output.write @theB
            while (true)
                @theA = get
                if (@theA == "/")
                    break
                elsif (@theA == "\\")
                    @output.write @theA
                    @theA = get
                elsif (@theA <= "\n")
                    raise "Unterminated RegExp Literal"
                end
                @output.write @theA
            end
            @theB = mynext
        end
    end
end

def get()
  c = @input.getc
  return EOF if(!c)
  c = c.chr
  return c if (c >= " " || c == "\n" || c.unpack("c") == EOF)
  return "\n" if (c == "\r")
  return " "
end

def isAlphanum(c)
   return false if !c || c == EOF
   return ((c >= 'a' && c <= 'z') || (c >= '0' && c <= '9') ||
           (c >= 'A' && c <= 'Z') || c == '_' || c == '$' ||
           c == '\\' || c[0] > 126)
end

def jsmin
    @theA = "\n"
    action(3)
    while (@theA != EOF)
        case @theA
        when " "
            if (isAlphanum(@theB))
                action(1)
            else
                action(2)
            end
        when "\n"
            case (@theB)
            when "{","[","(","+","-"
                action(1)
            when " "
                action(3)
            else
                if (isAlphanum(@theB))
                    action(1)
                else
                    action(2)
                end
            end
        else
            case (@theB)
            when " "
                if (isAlphanum(@theA))
                    action(1)
                else
                    action(3)
                end
            when "\n"
                case (@theA)
                when "}","]",")","+","-","\"","\\", "'", '"'
                    action(1)
                else
                    if (isAlphanum(@theA))
                        action(1)
                    else
                        action(3)
                    end
                end
            else
                action(1)
            end
        end
    end
end

def mynext()
    c = get
    if (c == "/")
        if(peek == "/")
            while(true)
                c = get
                if (c <= "\n")
                return c
                end
            end
        end
        if(peek == "*")
            get
            while(true)
                case get
                when "*"
                   if (peek == "/")
                        get
                        return " "
                    end
                when EOF
                    raise "Unterminated comment"
                end
            end
        end
    end
    return c
end

def peek()
    lookaheadChar = @input.getc
    @input.ungetc(lookaheadChar)
    return lookaheadChar.chr
end
