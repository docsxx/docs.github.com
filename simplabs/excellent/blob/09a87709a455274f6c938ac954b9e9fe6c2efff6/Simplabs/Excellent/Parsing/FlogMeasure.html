def process_block_pass(exp)
  add_flog_score(1)
  arg = exp.last
  if arg.is_a?(Sexp)
    case arg.first
      when :lit, :call
        add_flog_score(5)
      when :iter, :dsym, :dstr, *BRANCHES
        add_flog_score(10)
    end
  end
  super
end

def process_lit(exp)
  value = exp.shift
  case value
    when 0, -1
      # do nothing
    when Integer
      add_flog_score(0.25)
  end
  super
end
