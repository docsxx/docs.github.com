def process_default(exp, context = nil)
  @contexts.push(context) if context
  @contexts.each do |c|
    method = "process_#{exp.node_type}".to_sym
    c.send(method, exp) if c.respond_to?(method)
  end
  exp.children.each { |sub| process(sub) }
  apply_checks(exp)
  @contexts.pop if context
  exp
end
