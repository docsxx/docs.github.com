def method_missing(key, *args)
  return (@options[key.to_s.gsub(/\?$/, '').to_sym].eql?(true)) if key.to_s.match(/\?$/)
  if args.blank?
    @options[key.to_sym]
  else
    @options[key.to_sym] = args.size == 1 ? args.first : args
  end
end
