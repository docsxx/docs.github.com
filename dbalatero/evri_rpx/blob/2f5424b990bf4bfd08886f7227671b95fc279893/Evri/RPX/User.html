def initialize(json)
  if json.nil? or !json['profile']
    raise InvalidUserJsonError,
          'The JSON passed in is invalid!'
  end

  @json = json

  if @json['accessCredentials']
    @credentials = Credentials.new(@json['accessCredentials'])
  end
end
