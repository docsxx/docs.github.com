def increment_precision(objs)
  if objs.is_a?(SpecID::Pep) or objs.is_a?(SpecID::Prot)
    objs = [objs]
  end
  precision(objs)
end

def precision(objs)
  return 1.0 if objs.size == 0
  largest_q_value = objs.map {|v| v.q_value }.max
  prec = 1.0 - largest_q_value
end
