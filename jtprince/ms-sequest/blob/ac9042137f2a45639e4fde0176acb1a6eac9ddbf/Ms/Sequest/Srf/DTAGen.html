def from_io(fh)
  fh.pos = 0 if fh.pos != 0  
  st = fh.read(148)
  (@start_time, @start_mass, @end_mass, @num_dta_files, @group_scan, @min_group_count, @min_ion_threshold, @start_scan, @end_scan) = st.unpack('x36ex12ex4ex48Ix12IIIII')
  self
end
