def method_missing(method, *args, &block)
  if Parsers::Base.allowed_methods.include?(method)
    parser.send(method, *args, &block)
  else
    super
  end
end

def i_m_feeling_lucky(pattern, &block)
  if matches = match(pattern)
    m = matches[1]
    m = yield(m) if block_given?
    m
  end
end
