def node(key, &block)
  if block_given?
    value = ast[key]
    value = yield(value) unless value.nil?
    value
  else
    ast[key]
  end
end

def parse
  @input = StringScanner.new(@response.to_s)
  @ast = {}
  while !@input.eos?
    trim_newline  ||
    parse_content
  end
  @ast
end
