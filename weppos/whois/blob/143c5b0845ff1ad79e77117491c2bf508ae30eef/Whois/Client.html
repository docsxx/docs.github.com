def initialize(options = {}, &block)
  self.timeout = options[:timeout] || DEFAULT_TIMEOUT
  yield(self) if block_given?
end

def query(qstring)
  Timeout::timeout(timeout) do
    @server = Server.guess(qstring)
    @server.query(qstring)
  end
end
