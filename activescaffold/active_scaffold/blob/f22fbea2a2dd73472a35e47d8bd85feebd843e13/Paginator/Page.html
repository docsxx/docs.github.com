def initialize(pager, number, select) #:nodoc:
  @pager, @number = pager, number
  @offset = (number - 1) * pager.per_page
  @select = select
end

def last_item_number
  if next?
    @offset + @pager.per_page
  else
    @pager.count
  end
end
