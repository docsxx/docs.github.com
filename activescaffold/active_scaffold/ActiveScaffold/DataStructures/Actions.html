def exclude(*args)
  args.collect! { |a| a.to_sym } # symbolize the args
  @set.reject! { |m| args.include? m } # reject all actions specified
end

def swap(one, two)
  if include? one
    exclude one
    add two
  else
    exclude two
    add one
  end
end
