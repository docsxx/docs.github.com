def register(namespaces)
  @namespaces ||= {}
  @names ||= {}
  @prefixes ||= {}
  
  namespaces.each do |name, prefix|
    @prefixes[name.to_sym] = prefix.to_s
    @names[prefix.to_s] = name
    
    mod = Module.new
    meta = class << mod
      self
    end
    
    code_block = "def method_missing(sym, *a, &b)\nraise ::ArgumentError, \"Unexpected arguments for RDF::NS.expand\" if a.size > 0\n::RDF.expand_node(:\#{name}, sym.to_s)\nend\n\ndef const_missing(sym)\n::RDF.expand_node(:\#{name}, sym.to_s)\nend\n\ndef [](name)\n::RDF.expand_node(:\#{name}, name.to_s)\nend\n\n[:type, :name, :id].each{|a| private(a)}\n"
    
    meta.class_eval code_block
    
    @namespaces[name.to_sym] = mod
  end
end

def unregister(*names)
  names.each do |n|
    @names.delete(@prefixes[n])
    @prefixes.delete(n)
    @namespaces.delete(n)
  end
end

def unregister_all!
  defaults = [:rdf, :rdfs, :xsd, :owl]
  @namespaces.keys.each do |name|
    unregister(name) unless defaults.include?(name)
  end
end
