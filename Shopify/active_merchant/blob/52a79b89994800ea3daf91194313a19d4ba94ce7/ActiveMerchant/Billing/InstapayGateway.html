def authorize(money, creditcard, options = ;{})
  post = {}
  post[:authonly] = 1
  add_amount(post, money)
  add_invoice(post, options)
  add_creditcard(post, creditcard)
  add_address(post, options)
  add_customer_data(post, options)

  commit('ns_quicksale_cc', post)
end
