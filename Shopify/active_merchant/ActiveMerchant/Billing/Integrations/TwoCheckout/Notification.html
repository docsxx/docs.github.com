def status

  when 'Y'
    'Completed'
  when 'K'
    'Pending'
  else
    'Failed'
  end
end

def verify(secret)
  return false if security_key.blank?
  
  Digest::MD5.hexdigest("#{secret}#{params['sid']}#{transaction_id}#{gross}").upcase == security_key.upcase
end
