def guess(url)
  logger.info {"Guessing the repository type of #{url.inspect}"}
  
  handler = handlers.detect do |handler|
    logger.debug {"Asking #{handler}"}
    handler.understands_url?(url)
  end
  
  raise UnhandledUrl unless handler
  
  handler.new(url)
end
