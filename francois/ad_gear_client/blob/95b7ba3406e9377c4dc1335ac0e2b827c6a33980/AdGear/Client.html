def get(url, headers={})
  retried = false
  begin
    logger.debug {"GET #{url} #{headers.inspect}"}
    returning(RestClient.get(url, base_headers(headers))) do |response|
      logger.debug { response }
    end
  rescue RestClient::Unauthorized => e
    # Retry once only
    raise if retried

    logger.info "Failed GET to #{url}, retrying with correct authorization header"

    # TODO: Find a better way than mucking with the internals
    response_headers = e.response.instance_variable_get("@header")
    www_authenticate = response_headers["www-authenticate"].first

    header = AdGear::Digest.authenticate(url, @login, @password, www_authenticate)
    headers["Authorization"] = header

    retried = true
    retry
  end
end

def sites
  response = get(sites_url)
  doc = Hpricot::XML(response)
  (doc / "//site").inject([]) do |memo, site|
    params = site.children.inject({}) do |attrs, elem|
      next attrs if elem.name.strip.chomp.empty?
      attrs[elem.name.gsub("-", "_")] = elem.inner_html
      attrs
    end

    memo << Site.new(params)
  end
end
