def save
  retried = false
  headers = Hash.new
  begin
    uri = self.class.site.merge(URI.parse(self.class.collection_path))
    File.open(attributes["filename"]) do |file|
      params = {:upload => {:uploaded_data => file, :filename => File.basename(attributes["filename"])}}
      RestClient.post(uri.to_s, params, headers)
    end
  rescue RestClient::Unauthorized => e
    # Retry once only
    raise if retried

    www_authenticate = e.response["www-authenticate"]
    header = ActiveResource::Digest.authenticate(uri, self.class.user, self.class.password, www_authenticate, :post)
    headers["Authorization"] = header

    retried = true
    retry
  end
end
