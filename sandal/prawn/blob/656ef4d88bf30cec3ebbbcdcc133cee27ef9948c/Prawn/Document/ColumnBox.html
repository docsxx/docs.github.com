  @current_column = 0

def move_past_bottom
  @current_column = (@current_column + 1) % @columns
  @parent.y = @y
  if 0 == @current_column
    @parent.start_new_page
  end
end

def right_side
  columns_from_right = @columns - (1 + @current_column)
  absolute_right - (width_of_column * columns_from_right)
end
