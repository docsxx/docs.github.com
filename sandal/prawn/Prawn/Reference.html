  (@stream ||= "") << data  
end

def compress_stream
  @stream = Zlib::Deflate.deflate(@stream)
  @data[:Filter] = :FlateDecode
  @data[:Length] ||= @stream.length
  @compressed = true
end

def object 
  @on_encode.call(self) if @on_encode
  output = "#{@identifier} #{gen} obj\n" <<
           Prawn::PdfObject(data) << "\n"
  if @stream
    output << "stream\n" << @stream << "\nendstream\n" 
  end
  output << "endobj\n"
end
