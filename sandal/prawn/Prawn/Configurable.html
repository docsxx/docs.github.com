def configuration(*args)
  @config ||= Marshal.load(Marshal.dump(default_configuration))
  if Hash === args[0]
    @config.update(args[0])
  elsif args.length > 1
    @config.values_at(*args)
  elsif args.length == 1
    @config[args[0]]
  else
    @config
  end
end
