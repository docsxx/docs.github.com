def self.const_missing(const_name)
  if const_name == :MIME_TYPES
    hash = Hash.new { |hash,key| Rack::Mime::MIME_TYPES[".#{key}"] }
    const_set :MIME_TYPES, hash
    sinatra_warn 'Rack::File::MIME_TYPES is deprecated; use Rack::Mime instead.'
    hash
  else
    super
  end
end
