def frame_class(frame)
  if frame.filename =~ /lib\/sinatra.*\.rb/
    "framework"
  elsif (defined?(Gem) && frame.filename.include?(Gem.dir)) ||
        frame.filename =~ /\/bin\/(\w+)$/
    "system"
  else
    "app"
  end
end
