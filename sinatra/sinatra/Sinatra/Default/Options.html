def method_missing(name, *args, &block)
  if target.respond_to?(name)
    target.__send__(name, *args, &block)
  elsif args.empty? && name.to_s !~ /=$/
    sinatra_warn 'accessing undefined options will raise a NameError in Sinatra 1.0'
    nil
  else
    super
  end
end
