def &(other)
  i = 0
  o = ''
  while i < string.size
    if (c = string[i,1]) == other[i,1]
      o << c
    else
      o << ESC
    end
    i += 1
  end
  self.class.new(o)
end

def +(other)
  i = 0
  o = ''
  while i < string.size
    if other[i,1] == ESC
      o << string[i,1]
    else
      o << other[i,1]
    end
    i += 1
  end
  self.class.new(o)
end

def ^(other)
  i = 0
  o = ''
  while i < string.size
    if string[i,1] == ESC
      o << other[i,1]
    elsif other[i,1] == ESC
      o << string[i,1]
    else
      o << ESC
    end
    i += 1
  end
  self.class.new(o)
end

def apply(s=nil,*a,&b)
  if s
    string.send(s,*a,&b).to_mask
  else
    @_self ||= Functor.new do |op, *a|
      string.send(op,*a).to_mask
    end
  end
end
