status

public status(status = nil)

Nil#status makes it possible to pass messages through a "failure" chain. 

  def foo
    return nil.status("failed foo!")
  end

  result = foo
  if result.nil?
    result.status?  #=> true
    result.status   #=> "failed foo!"
  end
