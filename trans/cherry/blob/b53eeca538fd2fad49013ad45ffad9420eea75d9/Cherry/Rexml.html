def after( item )
  case item
  when object_class
    @node.parent.insert_after( @node, item.node )
  when Array, Elements
    item.each { |i| after( i ) }
  else
    @node.parent.insert_after @node, xml(item).node
  end
end

def empty!

  when ::REXML::Element

  when ::REXML::Text

  else
    nil
  end

def remove!

  when :text
    @node.value = ''
  else
    @node.parent.send("delete_#{@node.node_type}", @node )
  end
  self
end

def wrap( item )
  unless object_class === item
    item = xml(item)
  end
  @node.parent.insert_after @node, item.node
  item.append @node
  remove!
  item
end
