def after(item)
  case item
  when object_class
    @node.sbiling( item.node )
  when Array, Elements
    item.each { |i| after( i ) }
  else
    @node.sibling xml(item).node
  end
end

def before(item)
  case item
  when object_class
    @node.parent.insert_before( @node, item.node )
  when Array, Elements
    item.reverse_each { |i| before( i ) }
  else
    @node.parent.insert_before @node, xml(item).node
  end
end

def wrap( item )
  unless object_class === item
    item = xml(item)
  end
  @node.sibling item.node
  item.append @node
  remove!
  item
end
