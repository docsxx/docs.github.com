def self.up(n)
  if n.kind_of?(Integer)
    return self.new(File.join(['..'] * n))
  else
    return n.map { |i| self.up(i) }
  end
end

def join(*names)
  names = names.map { |name| name.to_s }

  return self.class.new(File.join(self,*names))
end
