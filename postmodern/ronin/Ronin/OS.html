def OS.define(name)
  name = name.to_s
  method_name = name.snake_case

  meta_def(method_name) do
    OS.new(:name => name)
  end

  meta_def("#{method_name}_version") do |version|
    OS.first_or_create(:name => name, :version => version.to_s)
  end

  return nil
end

def to_s
  if self.version
    return "#{self.name} #{self.version}"
  else
    return self.name.to_s
  end
