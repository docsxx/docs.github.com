def ParamParser.each_format(&block)
  ParamParser.formats.each do |format|
    block.call(format[:pattern],format[:parser])
  end
end

def ParamParser.recognize(pattern,&block)
  ParamParser.formats.unshift({
    :pattern => pattern,
    :parser => block
  })
end
