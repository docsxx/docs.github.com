def initialize(name,&block)
  @name = name.to_s

  defaults()

  Options.new(@name) do |opts|
    define_options(opts)

    @options = opts
  end

  block.call(self) if block
end

def self.run(*args)
  name = File.basename($0)
  name.gsub!(/^ronin-/,'')

  cmd = self.new(name)

  begin
    cmd.arguments(*(cmd.options.parse(args)))
  rescue OptionParser::MissingArgument,
         OptionParser::InvalidOption => e
    STDERR.puts e
    exit -1
  end

  return true
end
