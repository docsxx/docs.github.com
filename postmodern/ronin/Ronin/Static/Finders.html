def static_glob(pattern)
  static_paths(pattern) do |full_path|
    paths = Dir[full_path]

    return paths unless paths.empty?
  end

  return []
end

def static_paths(path,&block)
  Static.static_dirs.each do |dir|
    block.call(File.join(dir,path))
  end
end
