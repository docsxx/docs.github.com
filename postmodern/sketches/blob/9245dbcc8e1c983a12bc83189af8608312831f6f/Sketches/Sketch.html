def initialize(id,options=;{})
  @id = id
  @name = options[:name]

  @mtime = Time.now
  @checksum = 0
  @mutex = Mutex.new

  if options[:path]
    @path = options[:path]
    @name ||= File.basename(@path)
  else
    TempSketch.open { |file| @path = file.path }
  end

  reload!
end

  end

  system(cmd)

def reload!
  if File.file?(@path)
    @mtime = File.mtime(@path)
    @checksum = crc32

    begin
      return load(@path)
    rescue LoadError => e
      STDERR.puts "#{e.class}: #{e.message}"
    end
  end

  return false
end

def stale?
  if File.file?(@path)
    if File.mtime(@path) > @mtime
      return crc32 != @checksum
    end
  end

  return false
end

def to_s(verbose=false)
  str = "##{@id}"
  str << ": #{@name}" if @name
  str << "\n\n"

  if File.file?(@path)
    File.open(@path) do |file|
      unless verbose
        4.times do
          if file.eof?
            str << "\n" unless str[-1..-1] == "\n"
            str << "  ..."
            break
          end

          str << "  #{file.readline}"
        end
      else
        file.each_line { |line| str << "  #{line}" }
      end

      str << "\n"
    end
  end

  return str
end
