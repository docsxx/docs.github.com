def self.parse(args)
  @@opts = parse_args(args)
  if args.length > 0
    return args.sniff
  else
    return Reek::Sniffer.new($stdin.to_reek_source('$stdin'))
  end
end

def self.parse_args(args)
  result = default_options
  parser = OptionParser.new { |opts| set_options(opts, result) }
  parser.parse!(args)
  result
end

def self.set_options(opts, config)
  opts.banner = "Usage: \#{opts.program_name} [options] files...\n\nIf no files are given, Reek reads source code from standard input.\nSee http://wiki.github.com/kevinrutherford/reek for detailed help.\n"
  
  opts.separator "\nOptions:"
  set_all_options(opts, config)
end
