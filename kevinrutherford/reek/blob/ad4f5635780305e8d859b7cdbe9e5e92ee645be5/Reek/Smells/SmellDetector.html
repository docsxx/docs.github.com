def examine(context)
  before = @smells_found.size
  examine_context(context) if enabled? and !exception?(context)
  @smells_found.length > before
end

def report_on(report)
  @smells_found.each do |smell|
    if @masked
      report.record_masked_smell(smell)
    else
      report << smell
    end
  end
end
