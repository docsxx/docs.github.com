def full_report
  return quiet_report if Options[:quiet]
  result = header
  result += ":\n#{smell_list}" if should_report
  result += "\n"
  result
end

def smell_list
  smells = Options[:show_all] ? @all_warnings : @warnings
  smells.map {|smell| "  #{smell.report}"}.join("\n")
end
