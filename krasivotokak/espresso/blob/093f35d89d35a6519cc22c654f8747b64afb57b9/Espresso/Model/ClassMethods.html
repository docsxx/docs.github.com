def paginate_found(page = nil, query = nil, simple_query = nil)
  query ||= {}
  query.merge(parse_simple_query(simple_query)) if simple_query.present?
  @search = search(query)
  @results = @search.paginate(:page => page)
  [@search, @results]
end
