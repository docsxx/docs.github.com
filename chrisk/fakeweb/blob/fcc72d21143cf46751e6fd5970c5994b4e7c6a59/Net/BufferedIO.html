def initialize(io, debug_output = nil)
  @read_timeout = 60
  @rbuf = ''
  @debug_output = debug_output

  @io = case io
  when Socket, OpenSSL::SSL::SSLSocket, IO
    io
  when String
    if !io.include?("\0") && File.exists?(io) && !File.directory?(io)
      File.open(io, "r")
    else
      StringIO.new(io)
    end
  end
  raise "Unable to create local socket" unless @io
end
