def can?(actor, verb, *args)
  result = @klass.credential_options[:allow_by_default] || false
  @rules.each do |rule|
    result = true if rule.allow?(actor, verb, *args)
    result = false if rule.deny?(actor, verb, *args)
  end
  result
end
