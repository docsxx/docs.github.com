def method_to_js(method, *args)
  method_name = method.to_s 
  if method_name =~ /\?$/
    "#{self.var}.#{method_name.chop.to_js_method}(#{args.to_js_arguments})".to_sym
  else
    self << "#{self.var}.#{method.to_js_method}(#{args.to_js_arguments});"
  end
end

def open(options = {})
  options.default! :location => :center
  location = Google::OptionsHelper.to_location(options[:location])
  
  self << "center = #{self.var}.getCenter();" if location == :center
  
  self << "#{self.var}.openInfoWindowTabs(#{location}, tabs);"
end
