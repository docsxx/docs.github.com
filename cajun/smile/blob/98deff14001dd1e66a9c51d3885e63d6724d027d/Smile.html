def base_feed( options={} )
  options.merge!( :format => 'rss' )
  url = "http://api.smugmug.com/hack/feed.mg?"
  url_params =[]
  options.each_pair do |k,value|
    case k.to_s.downcase.to_sym
      when :popular_category
        key = :popularCategory
      when :geo_all 
        key = :geoAll
      when :geo_keyword 
        key = :geoKeyword
      when :geo_search 
        key = :geoSearch
      when :geo_community
        key = :geoCommunity 
      when :open_search_keyword
        key = :openSearchKeyword
      when :user_keyword 
        key = :userkeyword
      when :nickname_recent 
        key = :nicknameRecent
      when :nickname_popular 
        key = :nicknamePopular
      when :user_comments 
        key = :userComments
      when :geo_user 
        key = :geoUser
      when :geo_album 
        key = :geoAlbum
      when :size
        key = :Size
        value = value.titlecase
      when :image_count
        key = :ImageCount
      else
        key = k
    end
    
    url_params << "#{key.to_s}=#{ CGI.escape( value ) }"
  end
  
  content = RestClient.get( url + url_params.join( "&" ) )
  RSS::Parser.parse( content, false )
end
