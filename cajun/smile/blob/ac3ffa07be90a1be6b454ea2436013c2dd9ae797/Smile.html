def search( data, options={} )
  rss = search_rss( data, options )
  
  rss.items.map do |item| 
    image_id, image_key = item.link.split('/').last.split('#').last.split('_')
    Smile::Photo.find( :image_id => image_id, :image_key => image_key )
  end
end
