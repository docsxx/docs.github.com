Dynamic Method Handling

This class handles dynamic methods through the method_missing method in the class DataMapper::Collection

private method_missing(method, *args, &block)

Delegates to Model, Relationships or the superclass (LazyArray)

When this receives a method that belongs to the Model the Collection is scoped to, it will execute the method within the same scope as the Collection and return the results.

When this receives a method that is a relationship the Model has defined, it will execute the association method within the same scope as the Collection and return the results.

Otherwise this method will delegate to a method in the superclass (LazyArray) and return the results.

def method_missing(method, *args, &block)
  if model.respond_to?(method)
    delegate_to_model(method, *args, &block)
  elsif relationship = relationships[method] || relationships[method.to_s.singular.to_sym]
    delegate_to_relationship(relationship, *args)
  else
    super
  end
end

def create(attributes = {})
  if last_relationship.respond_to?(:resource_for)
    resource = super
    if create_intermediary(last_relationship => resource)
      resource
    end
  else
    if intermediary = create_intermediary
      super(attributes.merge(last_relationship.inverse => intermediary))
    end
  end
end

def destroy!
  unless intermediaries.destroy!
    return false
  end

  if empty?
    return true
  end

  # FIXME: use a subquery to do this more efficiently in the future
  repository_name = relationship.target_repository_name
  model           = relationship.target_model
  key             = model.key(repository_name)

  # TODO: handle compound keys
  model.all(:repository => repository_name, key.first => map { |r| r.key.first }).destroy!

  each { |r| r.reset }
  clear

  true
end

def save
  resources = if loaded?
    entries
  else
    head + tail
  end

  # delete only intermediaries linked to the target orphans
  unless intermediaries.all(last_relationship => @orphans).destroy
    return false
  end

  if last_relationship.respond_to?(:resource_for)
    super
    resources.all? { |r| create_intermediary(last_relationship => r) }
  else
    if intermediary = create_intermediary
      inverse = last_relationship.inverse
      resources.map { |r| inverse.set(r, intermediary) }
    end

    super
  end
end
