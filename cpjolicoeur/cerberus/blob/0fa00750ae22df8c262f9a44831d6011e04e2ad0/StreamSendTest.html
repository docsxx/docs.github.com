def setup
  @tmpfile = Tempfile.new("StreamSendTest")
  @tmpfilepath = @tmpfile.path()
  @tmpfile.unlink
  @servlisten = UNIXServer.new(@tmpfilepath)
  thServer = Thread.new { @server = @servlisten.accept }
  @iostream = UNIXSocket.new(@tmpfilepath)
  @stream = Stream.new
  @stream.start(@iostream)

  thServer.join
end
