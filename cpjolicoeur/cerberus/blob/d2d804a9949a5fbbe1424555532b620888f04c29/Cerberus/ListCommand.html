def run
  projects = Dir["#{HOME}/config/*.yml"]
  if projects.empty?
    puts "There are no any active projects" 
  else
    puts "List of active projects:"
    
    projects.each do |fn|
      fn =~ %r{#{HOME}/config/(.*).yml}
      
      puts "  * #{$1}"
    end
    
    puts "\nType 'cerberus build PROJECT_NAME' to build any of these projects"
  end
end
