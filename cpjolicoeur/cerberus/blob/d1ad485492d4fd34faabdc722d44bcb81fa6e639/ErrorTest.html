def test_type
  e = ErrorResponse.new
  assert_nil(e.type)
  e.type = :auth
  assert_equal(:auth, e.type)
  e.type = :cancel
  assert_equal(:cancel, e.type)
  e.type = :continue
  assert_equal(:continue, e.type)
  e.type = :modify
  assert_equal(:modify, e.type)
  e.type = :wait
  assert_equal(:wait, e.type)
  e.type = nil
  assert_nil(e.type)
end
