def to_form
  "Content-Disposition: form-data; name=\"#{self.name}\"; filename=\"#{@filename}\"\r\n" +
  "Content-Type: #{self.mime_type}\r\n" +
  "\r\n" +
  "#{self.value}\r\n"
end
