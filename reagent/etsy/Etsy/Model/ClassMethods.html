def finder(type, endpoint)
  parameter = endpoint.scan(/:\w+/).first
  parameter.sub!(/^:/, '')

  endpoint.sub!(":#{parameter}", '#{' + parameter + '}')

  send("find_#{type}", parameter, endpoint)
end
