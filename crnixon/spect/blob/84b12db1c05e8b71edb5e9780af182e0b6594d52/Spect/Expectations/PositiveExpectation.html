def method_missing(name, *args, &block)        
  if @object.respond_to?("#{name}?")
    assert @object.send("#{name}?", *args),
      "#{name}? expected to be true. #{@message}"
  else
    if @object.respond_to?(name)
      assert @object.send(name, *args),
        "#{name} expected to be true. #{@message}"
    else
      super
    end
  end
end
