def find_me(arg)
  if @names[arg]
    return [true,nil]
  end
  @names.keys.each() do |name|
    if name =~ /^-(\w)$/
      match_string = "^\\-(\\w*)#{$1}(\\w*)$"
      match_data = arg.match(match_string)
      if match_data
        # Note that if [1] and [2] were both empty 
        # we'd have returned above
        return [true, "-" + match_data[1] + match_data[2]]
      end
    end
  end
  [false]
end

def get_value!(args)
  idx = -1
  args.each_index do |i|
    result = find_me(args[i])
    if result[0]
      if result[1]
        args[i] = result[1]
      else
        args.delete_at i
      end
      return result[0]
    end
  end
  false
end
