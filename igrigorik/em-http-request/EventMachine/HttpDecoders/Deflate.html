def decompress(compressed)
  begin
    @zstream ||= Zlib::Inflate.new(nil)
    @zstream.inflate(compressed)
  rescue Zlib::Error
    raise DecoderError
  end
end

def finalize
  return nil unless @zstream

  begin
    r = @zstream.inflate(nil)
    @zstream.close
    r
  rescue Zlib::Error
    raise DecoderError
  end
end
