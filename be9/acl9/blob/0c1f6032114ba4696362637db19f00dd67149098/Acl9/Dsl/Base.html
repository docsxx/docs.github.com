def allowance_expression
  allowed_expr = if @allows.size > 0
                   @allows.map { |clause| "(#{clause})" }.join(' || ')
                 else
                   "false"
                 end

  not_denied_expr = if @denys.size > 0
                      @denys.map { |clause| "!(#{clause})" }.join(' && ')
                    else
                      "true"
                    end

  [allowed_expr, not_denied_expr].
    map { |expr| "(#{expr})" }.
    join(default_action == :deny ? ' && ' : ' || ')
end
