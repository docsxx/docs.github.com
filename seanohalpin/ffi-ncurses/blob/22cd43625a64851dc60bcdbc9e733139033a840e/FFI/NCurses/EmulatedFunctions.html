def self.fixup(function, &block)
  if NCurses.unattached_functions.include?(function)
    block.call
  end
end

def setsyx(y, x)
  if y == -1 && x == -1
    leaveok(newscr, NCurses::TRUE)
  else
    leaveok(newscr, NCurses::FALSE)
    wmove(newscr, y, x)
  end
end
