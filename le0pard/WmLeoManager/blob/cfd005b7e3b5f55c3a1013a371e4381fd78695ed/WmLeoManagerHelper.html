def file_size_to_norm(size)
  size = size.to_f
  ext = wm_translate('bt')
  if size > 1024
    size = size / 1024
    ext = wm_translate('kbt')
  end
  if size > 1024
    size = size / 1024
    ext = wm_translate('mbt')
  end
  return sprintf("%0.2f", size) + " " + ext
end

def wm_translate(str)
  if @config && @language
    if @language[@config['conf']['lang']]
      if @language[@config['conf']['lang']][str]
        return @language[@config['conf']['lang']][str]
      else
        return "no translation"
      end
    else
      return "no translation"
    end
  else
    return "no translation"
  end
end
