def check_and_r_path(path,name)
  chars = "/"
  name = name.gsub(/^[#{chars}]+|[#{chars}]+$/, '')
  temp_path = File.expand_path(File.join(path, name))
  if temp_path.start_with?(@@main_path)
    return temp_path
  else
    return false
  end
end

def create_dir(path,filename)
  if self.check_and_r_path(path,filename)
    file = File.expand_path(File.join(path, filename))
    begin
      Dir.mkdir(file) unless File.exist?(file)
      return true
    rescue
       nil
    end
  end
  nil
end

def delete(path,name)
  if self.check_and_r_path(path, name)
    file_path = File.expand_path(File.join(path, name))
    begin
      if File.ftype(file_path) == 'file'
        File.unlink(file_path) if File.exist?(file_path)
      elsif File.ftype(file_path) == 'directory'
        Dir.rmdir(file_path) if File.exist?(file_path)
      end
      return true
    rescue
       nil
    end
  end
  nil
end

def file_path_to_web_path(file)
  web_path = File.expand_path(File.join(RAILS_ROOT, 'public'))
  file_temp = file.sub(web_path, '')
  file_array = file_temp.split(File::SEPARATOR)
  file_temp = file_array.join("/")
  return file_temp
end

def get_file_info(path, filename)
  allpath = self.check_and_r_path(path,filename)
  if allpath && File.exist?(allpath)
    if File.ftype(allpath) == 'file'
       
        file_data = {
          :name => filename,
          :path => allpath,
          :ftype => File.ftype(allpath),
          :ext_name => filename.gsub(/.*\./, '').to_s.downcase,
          :ext => File.extname(allpath),
          :size => File.size(allpath),
          :readable => File.readable?(allpath),
          :writable => File.writable?(allpath),
          :executable => File.executable?(allpath),
          :dir => File.dirname(allpath),
          :atime => File.atime(allpath),
          :mtime => File.mtime(allpath),
          :ctime => File.ctime(allpath),
          :web => self.file_path_to_web_path(allpath)
        }
        
        if file_data[:ext_name] == 'jpg' || 
          file_data[:ext_name] == 'jpeg' || 
          file_data[:ext_name] == 'bmp' ||
          file_data[:ext_name] == 'gif' || 
          file_data[:ext_name] == 'png'
          
          file_data[:is_image] = true
        end  
        
        if file_data[:readable] && file_data[:is_image]
          file_data[:image_type] = self.image_type(allpath)
        end
        
        return file_data
    end
  end
  nil
end

def get_file_list(path,name,options = {})
  allpath = self.check_and_r_path(path,name)
  if allpath && File.exist?(allpath)
    
    files = Array.new
    files_size = 0
    folder_count = 0
    file_count = 0
    current_root = false
    if self.check_current_root(allpath)
      current_root = true
    end
    
    Dir.entries(allpath).sort.each do |filename|
      i = files.length
      temp_file = File.join(allpath, filename)
      if File.ftype(temp_file) == 'file'
        fsize = File.size(temp_file)
        
        files[i] = {
          :name => filename,
          :path => temp_file,
          :ftype => File.ftype(temp_file),
          :ext_name => filename.gsub(/.*\./, '').to_s.downcase,
          :ext => File.extname(temp_file),
          :size => fsize,
          :readable => File.readable?(temp_file),
          :writable => File.writable?(temp_file),
          :executable => File.executable?(temp_file),
          :dir => File.dirname(temp_file),
          :atime => File.atime(temp_file),
          :mtime => File.mtime(temp_file),
          :ctime => File.ctime(temp_file)
        }
        
        if files[i][:ext_name] == 'jpg' || 
          files[i][:ext_name] == 'jpeg' || 
          files[i][:ext_name] == 'bmp' ||
          files[i][:ext_name] == 'gif' || 
          files[i][:ext_name] == 'png'
          
          files[i][:is_image] = true
        end
        
        files_size += fsize
        file_count += 1
      elsif File.ftype(temp_file) == 'directory'
          if filename == '..' || filename == '.'
            next
          end
          
          files[i] = {
            :name => filename,
            :path => temp_file,
            :ftype => File.ftype(temp_file),
            :ext_name => '',
            :ext => File.extname(temp_file),
            :size => 0,
            :readable => File.readable?(temp_file),
            :writable => File.writable?(temp_file),
            :executable => File.executable?(temp_file),
            :dir => temp_file,
            :atime => File.atime(temp_file),
            :mtime => File.mtime(temp_file),
            :ctime => File.ctime(temp_file)
          }
          folder_count += 1
      end
    end
    
    if options[:sort]
      if options[:sort] == 'r_name'
        files.sort! {|a,b| b[:name] <=> a[:name]}
      elsif options[:sort] == 'size'
        files.sort! {|a,b| a[:size] <=> b[:size]}
      elsif options[:sort] == 'r_size'
        files.sort! {|a,b| b[:size] <=> a[:size]}
      elsif options[:sort] == 'mtime'
        files.sort! {|a,b| a[:mtime] <=> b[:mtime]}
      elsif options[:sort] == 'r_mtime'
        files.sort! {|a,b| b[:mtime] <=> a[:mtime]}
      elsif options[:sort] == 'ftype'
        files.sort! {|a,b| a[:ftype] <=> b[:ftype]}
      elsif options[:sort] == 'r_ftype'
        files.sort! {|a,b| b[:ftype] <=> a[:ftype]}
      end
    end
    
    if !current_root
      filename = '..'
      temp_file = File.expand_path(File.join(allpath, filename))
      if File.exist?(temp_file)
        file = {
            :name => filename,
            :path => temp_file,
            :ftype => File.ftype(temp_file),
            :ext_name => '',
            :ext => File.extname(temp_file),
            :size => 0,
            :readable => File.readable?(temp_file),
            :writable => File.writable?(temp_file),
            :executable => File.executable?(temp_file),
            :dir => temp_file,
            :atime => File.atime(temp_file),
            :mtime => File.mtime(temp_file),
            :ctime => File.ctime(temp_file)
          }
         files.unshift(file)
      end
    end
    
    return {:files => files, 
            :files_size => files_size, 
            :folder_count => folder_count, 
            :file_count => file_count,
            :path => allpath,
            :options => options,
            :main_path => @@main_path}
  end
  nil
end

def image_type(file)
  case IO.read(file, 10)
    when /^GIF8/: 'gif'
    when /^\x89PNG/: 'png'
    when /^\xff\xd8\xff\xe0\x00\x10JFIF/: 'jpg'
    when /^\xff\xd8\xff\xe1(.*){2}Exif/: 'jpg'
  else 'unknown'
  end
end

def rename(path, old_file, new_file)
  if self.check_and_r_path(path, old_file)
    new_file = self.sanitize_filename(new_file)
    old_file_path = File.expand_path(File.join(path, old_file))
    new_file_path = File.expand_path(File.join(path, new_file))
    begin
      File.rename(old_file_path, new_file_path) if File.exist?(old_file_path)
      return true
    rescue
       nil
    end
  end
  nil
end

def upload(file_data,file_path,file_name)
  if file_data && file_path
    if (!file_data.original_filename.blank?)
      
      if ((@@config['conf']['check_files_type'] && @@config['conf']['files_type'].include?(file_data.content_type.chomp)) || !@@config['conf']['check_files_type'])
        if ((@@config['conf']['check_file_size'] && @@config['conf']['file_size'].to_i >= file_data.size.to_i) || !@@config['conf']['check_file_size'])
          if file_name.blank?
            file_name = file_data.original_filename.to_s
          end
          
          if self.check_and_r_path(file_path,file_name)
            File.open(File.expand_path(File.join(file_path, file_name)), File::CREAT|File::WRONLY) { |f| f.write(file_data.read) }
            return true
          end
        end
      end
    end
  end
  nil
end
