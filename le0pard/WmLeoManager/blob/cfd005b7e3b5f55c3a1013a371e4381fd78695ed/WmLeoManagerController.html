def create_dir
  if request.xhr? && params[:path] && params[:filename]
    filemanager = Filemanager.new(@config)
    if filemanager.create_dir(params[:path],params[:filename])
      render :text => ''
    else
      render :text => wm_translate('error_create_dir')
    end
  end
end

def delete
  if request.xhr? && params[:path] && params[:file]
    filemanager = Filemanager.new(@config)
    for fl in params[:file]
      filemanager.delete(params[:path],fl)
    end
    render :text => ''
  end
end

def download_file
  if params[:file]
    filemanager = Filemanager.new(@config)
    if filemanager.check_path(params[:file]) && File.exist?(params[:file])
      f = File.new(params[:file])
      text = f.read
      send_data text, :filename => File.basename(params[:file]), :disposition => 'attachment'
    end
  end
end

def get_file_list
  if request.xhr? && params[:path]
    options = {}
    if params[:sort]
      options[:sort] = params[:sort]
    end
    if params[:filename]
      filename = params[:filename]
    else
      filename = '.'
    end
    filemanager = Filemanager.new(@config)
    @files = filemanager.get_file_list(params[:path], filename, options)
    render :partial => 'shared/files_list'
  end
end

def rename
  if request.xhr? && params[:path] && params[:old_filename] && params[:new_filename]
    filemanager = Filemanager.new(@config)
    if params[:old_filename] != params[:new_filename]
      if filemanager.rename(params[:path],params[:old_filename],params[:new_filename])
        render :text => ''
      else
        render :text => wm_translate('error_rename')
      end
    else
      render :text => wm_translate('error_rename')
    end
  end
end

def upload_file
  if request.post?
    if params[:upload]
      if params[:upload][:file] && params[:upload][:upload_path]
        filemanager = Filemanager.new(@config)
        file_data = params[:upload][:file]
        file_path = params[:upload][:upload_path]
        file_name = nil
        if params[:upload][:name]
          file_name = params[:upload][:name]
        end
        
        if filemanager.upload(file_data,file_path,file_name)
          render :text => '<script type="text/javascript">window.parent.wmLeoManagerObject.stopUpload();</script>'
        else
          render :text => '<script type="text/javascript">alert(\'' + wm_translate('error_upload') + '\');</script>'
        end
      else
        render :text => '<script type="text/javascript">alert(\'' + wm_translate('error_upload') + '\');</script>'
      end
    end
  end
end
