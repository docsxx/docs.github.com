def assert_prefixes!(expression, ns)
  expression.scan(/([a-zA-Z_][a-zA-Z0-9_.-]*):/).map{|m| m[0] }.each do |prefix|
    raise "Undefined prefix '#{prefix}'" if ns[prefix].nil?
  end
end
