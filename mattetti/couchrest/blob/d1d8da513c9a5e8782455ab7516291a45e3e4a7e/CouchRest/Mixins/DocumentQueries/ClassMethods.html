def first(opts = {})
  first_instance = self.all(opts.merge!(:limit => 1))
  first_instance.empty? ? nil : first_instance.first
end

def get(id, db = database)
  begin
    doc = db.get id
  rescue
    nil
  else
    new(doc)
  end
end
