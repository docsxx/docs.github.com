Class: Rdb4o::Type

Type casting

To prevent errors like: class Foo 

  include Rdb4o::Model

  field :num, Fixnum

end

f = Foo.new f.num = "4" # => type error

def accessors_for(type, name)
  type = self.for(type)
  if type.respond_to?(:accessors)
    type.accessors(name)
  else
    Rdb4o::Types::Generic.accessors(name)
  end
end

def for(type)
    if type.is_a?(Array)
      Rdb4o::Types::Array.with(type.first)
    else
      type = Extlib::Inflection.demodulize(type.to_s)
      if Rdb4o::Types.constants.include?(type)
        Rdb4o::Types.const_get(type)
      else
        type
      end
    end
  end
