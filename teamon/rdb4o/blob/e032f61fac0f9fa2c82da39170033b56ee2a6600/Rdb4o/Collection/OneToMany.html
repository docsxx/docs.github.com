def initialize(parent, model, relation_name, foreign_key)
  super(model)
  @parent, @relation_name, @foreign_key = parent, relation_name, foreign_key
  conditions.merge!(@foreign_key => @parent)
end

def method_missing(method_name, *args, &proc)
  if scope = model.scopes[method_name]
    if scope.is_a?(Hash)
      all(scope)
    elsif scope.is_a?(Proc)
      all(&scope)
    end
  else
    super
  end
end
