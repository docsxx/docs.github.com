def dump_attributes!
  self.class.fields.each_pair do |name, field|
    send("set_#{name}", field.dump(attributes[name])) if respond_to? "set_#{name}"
  end
  self
end

def load_attributes!
  self.class.fields.each_pair do |name, field|
    attributes[name] = field.load(send("get_#{name}")) if respond_to? "get_#{name}"
  end
  self
end
