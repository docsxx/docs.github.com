def <<(photo)
  self.photos ||= []
  self.photos << photo if photo.is_a?(Flickr::Photos::Photo)
end

def next_page
  api.send(self.method, options.merge(:page => self.page.to_i + 1)) if self.page.to_i < self.pages.to_i
end
