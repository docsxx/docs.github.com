def get_portrait_url(user_id, options={})
  size = options[:size]

  sig_options = {
    :user_id => user_id,
    :method => "vimeo.people.getPortraitUrl"
  }
  sig_options.merge! :size => size unless size.nil?

  make_request sig_options
end
