Returns a hash representation of the object. Calls #to_hash on its contents (which should be station groups) and then merges those hashes. The end results should look something like: 

 {"Closed stations" => [station group contents...],

def to_xml( as_string=true )
  doc = REXML::Document.new
  root = doc.add_element( "station_groups" )
  each do |e|
    root.add_element( e.to_xml( false ) )
  end
  if as_string then doc.to_s else doc end
end
