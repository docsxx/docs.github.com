def user_name(name = nil)
  return name.scan(/\w+/).shuffle.join(%w(. _).rand).downcase if name
  
  [ 
    Proc.new { Name.first_name.gsub(/\W/, '').downcase },
    Proc.new { 
      [ Name.first_name, Name.last_name ].map {|n| 
        n.gsub(/\W/, '')
      }.join(%w(. _).rand).downcase }
  ].rand.call
end
