Class: Haml::HTML

This class contains the functionality used in the html2haml utility, namely converting HTML documents to Haml templates. It depends on Hpricot for HTML parsing (http://code.whytheluckystiff.net/hpricot/).

def initialize(template, options = ;{})
  @options = options

  if template.is_a? Hpricot::Node
    @template = template
  else
    if template.is_a? IO
      template = template.read
    end

    if @options[:rhtml]
      match_to_html(template, /<%=(.*?)-?%>/m, 'loud')
      match_to_html(template, /<%-?(.*?)-?%>/m,  'silent')
    end

    method = @options[:xhtml] ? Hpricot.method(:XML) : method(:Hpricot)
    @template = method.call(template.gsub('&', '&amp;'))
  end
end
