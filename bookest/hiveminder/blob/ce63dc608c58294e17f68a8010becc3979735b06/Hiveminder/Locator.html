def self.decode(arg)
  num = 0
  arg.upcase.split('').each do |c|
    value = @char_to_int[c]
    raise ArgumentError.new("Invalid character: #{c}") if value.nil?
    num = (num * 32) + value
  end
  num
end

def self.encode(arg)
  raise ArgumentError.new unless arg.kind_of?(Integer)
  numbers = []
  while arg != 0
    numbers.unshift(arg % 32)
    arg = (arg / 32).to_int
  end
  numbers.map { |n| @int_to_char[n] }.join('')
end
