def initialize(args = ;{})
  @server = args[:server] || "127.0.0.1"
  @port = args[:port] || "8021"
  @auth = args[:auth] || "ClueCon"
  @socket = TCPSocket.new(@server, @port)
  super(@socket)
  # Attempt to login or raise an exception
  unless login
    raise "Unable to login, check your password!"
  end
end
