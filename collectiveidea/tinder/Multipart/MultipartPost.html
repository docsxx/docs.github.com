def prepare_query()
  @query = @params.map do |k,v|
    param = v.respond_to?(:read) ? FileParam.new(k, v.path, v.read) : Param.new(k, v)
    "--#{BOUNDARY}\r\n#{param.to_multipart}"
  end.join("") + "--#{BOUNDARY}--"
end
