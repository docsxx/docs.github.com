def to_s
  ([ "Content-Disposition: form-data" ] +
  attributes.map{|k,v| "#{k}=\"#{v}\""}).
  join('; ') + "\r\n"+
  (@mime_type ? "Content-Type: #{@mime_type}\r\n" : '')+
  "\r\n#{data}"
end
