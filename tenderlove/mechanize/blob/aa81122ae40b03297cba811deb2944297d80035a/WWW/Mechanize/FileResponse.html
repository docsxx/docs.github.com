def [](key)
  return nil unless key.downcase == 'content-type'
  return 'text/html' if directory?
  return 'text/html' if ['.html', '.xhtml'].any? { |extn|
    @file_path =~ /#{extn}$/
  }
  nil
end

def content_length

def read_body
  if ::File.exists?(@file_path)
    if directory?
      yield dir_body
    else
      yield ::File.read(@file_path)
    end
  else
    yield ''
  end
end
