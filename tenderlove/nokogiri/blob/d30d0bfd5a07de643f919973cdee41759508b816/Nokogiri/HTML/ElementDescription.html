def self.[](tag_name) # :nodoc:
  ptr = LibXML.htmlTagLookup(tag_name)
  return nil if ptr.null?

  desc = allocate
  desc.cstruct = LibXML::HtmlElemDesc.new(ptr)
  desc
end
