def add_decorators(doc)
  doc.decorators(XML::Node) << Decorators::Hpricot::Node
  doc.decorators(XML::NodeSet) << Decorators::Hpricot::NodeSet
  doc.decorate!
  doc
end

def make string
  doc = XML::Document.new
  ns = XML::NodeSet.new(doc)
  ns << XML::Text.new(string, doc)
  ns
end
