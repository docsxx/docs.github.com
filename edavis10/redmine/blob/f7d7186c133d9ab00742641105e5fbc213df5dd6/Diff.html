def initialize(diffs_or_a, b = nil, isstring = nil)

  else

    @curdiffs = []

def Diff.lcs(a, b)
  astart = 0
  bstart = 0
  afinish = a.length-1
  bfinish = b.length-1
  mvector = []
  
  # First we prune off any common elements at the beginning
  while (astart <= afinish && bstart <= afinish && a[astart] == b[bstart])
    mvector[astart] = bstart
    astart += 1
    bstart += 1
  end
  
  # now the end

    mvector[afinish] = bfinish
    afinish -= 1
    bfinish -= 1
  end
