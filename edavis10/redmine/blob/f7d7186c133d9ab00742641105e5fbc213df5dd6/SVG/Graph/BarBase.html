def initialize config
  raise "fields was not supplied or is empty" unless config[:fields] &&
  config[:fields].kind_of?(Array) &&
  config[:fields].length > 0
  super
end

def min_value
  min = 0
  if min_scale_value.nil? 
    min = @data.collect{|x| x[:data].min}.min
    min = min > 0 ? 0 : min
  else
    min = min_scale_value
  end
  return min
end
