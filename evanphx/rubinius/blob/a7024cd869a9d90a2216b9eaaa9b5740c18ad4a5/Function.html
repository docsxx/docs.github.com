def values(x) # <= defines functions solved
  f = []
  f1 = x[0]*x[0] + x[1]*x[1] - @two # f1 = x**2 + y**2 - 2 => 0
  f2 = x[0] - x[1]                  # f2 = x    - y        => 0
  f <<= f1
  f <<= f2
  f
end
