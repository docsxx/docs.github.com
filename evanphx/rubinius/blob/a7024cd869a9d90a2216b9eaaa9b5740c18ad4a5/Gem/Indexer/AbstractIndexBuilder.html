Class: Gem::Indexer::AbstractIndexBuilder

Abstract base class for building gem indicies. Uses the template pattern with subclass specialization in the begin_index, end_index and cleanup methods.

def build
  FileUtils.mkdir_p @directory unless File.exist? @directory
  raise "not a directory: #{@directory}" unless File.directory? @directory

  file_path = File.join @directory, @filename

  @files << file_path

  File.open file_path, "wb" do |file|
    @file = file
    start_index
    yield
    end_index
  end
  cleanup
ensure
  @file = nil
end

def compress(filename, ext="rz")
  zipped = zip(File.open(filename, 'rb'){ |fp| fp.read })
  File.open "#{filename}.#{ext}", "wb" do |file|
    file.write zipped
  end
end
