def bind(cl)
  self.module_eval %[
    def Raise(err = nil, *rest)
      Exception2MessageMapper.Raise(self.class, err, *rest)
    end
    alias Fail Raise

    def self.included(mod)
      mod.extend Exception2MessageMapper
    end
  ]
end
