def setup 
  @@setup and return

  module_eval {
    # a hack to shut up warning
    alias old_init initialize
    remove_method :old_init
  }
  begin
  require 'rbtree'

  module_eval %{
def initialize(*args, &block)
  @hash = RBTree.new
  super
end
  }
  rescue LoadError
  module_eval %{
def initialize(*args, &block)
  @keys = nil
  super
end

def clear
  @keys = nil
  super
end

def replace(enum)
  @keys = nil
  super
end

def add(o)
  @keys = nil
  @hash[o] = true
  self
end
alias << add

def delete(o)
  @keys = nil
  @hash.delete(o)
  self
end

def delete_if
  n = @hash.size
  @hash.delete_if { |o,| yield(o) }
  @keys = nil if @hash.size != n
  self
end

def merge(enum)
  @keys = nil
  super
end

def each
  to_a.each { |o| yield(o) }
end

def to_a
  (@keys = @hash.keys).sort! unless @keys
  @keys
end
  }
  end

  @@setup = true
end
