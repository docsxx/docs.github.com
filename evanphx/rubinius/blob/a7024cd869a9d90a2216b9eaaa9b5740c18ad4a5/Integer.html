def Integer.from_prime_division(pd)
  value = 1
  for prime, index in pd
    value *= prime**index
  end
  value
end

def downto(val)
  return to_enum :downto, val unless block_given?
  i = self
  while i >= val
    yield i
    i -= 1
  end
  self
end

def gcd(other)
  min = self.abs
  max = other.abs
  while min > 0
    tmp = min
    min = max % min
    max = tmp
  end
  max
end

def gcd2(int)
  a = self.abs
  b = int.abs
  a, b = b, a if a < b
  
  pd_a = a.prime_division
  pd_b = b.prime_division
  
  gcd = 1
  for pair in pd_a
    as = pd_b.assoc(pair[0])
    if as
gcd *= as[0] ** [as[1], pair[1]].min
    end
  end
  return gcd
end

def gcdlcm(other)
  gcd = self.gcd(other)
  if self.zero? or other.zero?
    [gcd, 0]
  else
    [gcd, (self.div(gcd) * other).abs]
  end
end

def lcm(other)
  if self.zero? or other.zero?
    0
  else
    (self.div(self.gcd(other)) * other).abs
  end
end

def prime_division
  raise ZeroDivisionError if self == 0
  ps = Prime.new
  value = self
  pv = []
  for prime in ps
    count = 0
    while (value1, mod = value.divmod(prime)
     mod) == 0
value = value1
count += 1
    end
    if count != 0
pv.push [prime, count]
    end
    break if prime * prime  >= value
  end
  if value > 1
    pv.push [value, 1]
  end
  return pv
end
