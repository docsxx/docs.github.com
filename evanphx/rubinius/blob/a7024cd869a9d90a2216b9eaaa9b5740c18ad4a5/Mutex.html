def lock
  @lock.receive
  begin
    while @owner
      wchan = Rubinius::Channel.new
      @waiters.push wchan
      @lock << nil
      wchan.receive
      @lock.receive
    end
    @owner = Thread.current
    self
  ensure
    @lock << nil
  end
end

def try_lock
  @lock.receive
  begin
    if @owner
      false
    else
      @owner = Thread.current
      true
    end
  ensure
    @lock << nil
  end
end

def unlock
  @lock.receive
  begin
    if @owner != Thread.current
      raise ThreadError, "Not owner, #{@owner.inspect} is"
    end

    @owner = nil
    @waiters.shift << nil unless @waiters.empty?
    self
  ensure
    @lock << nil
  end
end
