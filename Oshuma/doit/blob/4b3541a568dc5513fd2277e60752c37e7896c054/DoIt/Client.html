def run
  input = OptionParser.new do |opts|
    opts.banner = "Usage: #{File.basename($0)} [options]"
    opts.on("-a", "--all", "Show even completed tasks.") do |a|
      @options[:all] = a
    end
    opts.on("--debug", "Turn on debugging output.") do |debug|
      $DEBUG = @options[:debug] = debug
    end
    opts.on("-d <id>", "--delete", "Delete a given Task.") do |id|
      @options[:delete] = id
    end
    opts.on("-e <id>", "--edit", "Edit a given Task.") do |id|
      @options[:edit] = id
    end
    opts.on("-f <format>", "--format", "Valid formats: text, json, xml") do |format|
      @options[:format] = format
    end
    opts.on("-n [task]", "--new", "Create a new Task.") do |task|
      new_task(task)
    end
    opts.on("-p <path>", "--path", "Specify the database URI to use.") do |path|
      @options[:path] = path
      ENV['DOIT_DB'] = path
    end
    opts.on("-r", "--report", "Show list of tasks.") do |r|
      @options[:report] = r
    end
    opts.on("-s", "--sync", "Sync it up!") do |s|
      @options[:sync] = s
    end
    opts.on("-t <id>", "--toggle", "Toggle the 'completed' flag of the Task.") do |id|
      @options[:toggle] = id
    end
    opts.on("-v", "--version", "Show the version.  No, really!") do |v|
      output(DoIt.to_version)
      exit
    end
    opts.on("-y", "--yes", "Don't prompt for confirmation.") do |confirm|
      @options[:confirm] = confirm
    end
  end

  begin
    @argv << '--help' if @argv.empty?
    input.parse!(@argv)
    take_action!
  rescue OptionParser::InvalidOption => error
    debug(error)
    error("#{error}\nTry passing '--help'")
  rescue OptionParser::MissingArgument => error
    debug(error)
    error("#{error}\nTry passing '--help'")
  end
end
