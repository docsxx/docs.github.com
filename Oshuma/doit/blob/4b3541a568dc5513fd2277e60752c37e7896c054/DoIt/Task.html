def self.database(&block)
  DataMapper.setup(:default, "sqlite3://#{DB}")
  DataMapper.auto_upgrade!
  raise 'Must pass a block.' unless block_given?
  yield
end

def self.to_json(find_all = false)
  order = [:created_at.asc]
  tasks = database do
    if find_all
      all(:order => order)
    else
      all(:order => order, :completed_at => nil)
    end
  end
  $stdout.puts tasks.inspect if $DEBUG
  $stdout.puts(tasks.empty? ? 'No tasks.' : tasks.to_json)
end

def self.to_txt(find_all = false)
  order = [:created_at.asc]
  tasks = database do
    if find_all
      all(:order => order)
    else
      all(:order => order, :completed_at => nil)
    end
  end
  $stdout.puts tasks.inspect if $DEBUG
  if tasks.empty?
    $stdout.puts 'No tasks.'
    return
  end
  tasks.each do |task|
    header  = "(#{task.id}) | #{task.created_at}"
    $stdout.puts header
    $stdout.puts "--- COMPLETED: #{task.completed_at}" if task.completed_at
    $stdout.puts '-' * header.size
    $stdout.puts task.content
    $stdout.puts # newline
  end
end

def self.toggle(id)
  database do
    task = get(id.to_i)
    $stdout.puts task.inspect if $DEBUG
    if task.completed_at
      task.update_attributes(:completed_at => nil)
    else
      task.update_attributes(:completed_at => Time.now)
    end
  end
end
