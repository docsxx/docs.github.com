def pick_with_index!
  tokens = to_a
  s, index = tokens.pick_with_index
  start = tokens[0, index].join
  self[start.size, s.size] = ""
  [s, index]
end

def shuffle_chars!
  (0...size).each {|j| 
    i = Kernel.rand(size-j)
    self[j], self[j+i] = self[j+i], self[j]
  }
  self
end
