def method_missing(method, *params)
  property = method.to_s
  if property =~ /(.*)=$/  # It's a value assignment
    @newvalues ||= []
    @newvalues << $1
    @properties[$1] = clean_value(params[0])
   else
    if @properties.has_key?(property)
      if self.class.boolean_fields.include?(property)
        (@properties[property] == 1)
      else
        @properties[property]
      end
    else
      super
    end
  end
end

def subscribe(contact)
  c = ActsAsIcontact::Contact.find(contact)
  s = ActsAsIcontact::Subscription.new(:contactId => c.id, :listId => id)
  s.save
end
