def initialize(data = nil)
  @uid = data ? data[:uid] : nil
  @created = data ? data[:created] : Time.now
  @modified = data ? data[:modified] : @created
  @path = data ? data[:path] : nil
  @name = data ? data[:name] : nil
  @connection = nil
end
