def delete(query)
  records = @records[query.model]

  read_many(query).each do |resource|
    records.delete(resource)
  end.size
end

def update(attributes, query)
  read_many(query).each do |resource|
    attributes.each do |property,value|
      property.set!(resource, value)
    end
  end.size
end
