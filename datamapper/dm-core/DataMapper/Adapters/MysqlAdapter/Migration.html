def field_exists?(storage_name, field_name)
  statement = "SELECT COUNT(*)\nFROM `information_schema`.`columns`\nWHERE `table_schema` = ?\nAND `table_name` = ?\nAND `column_name` = ?\n".compress_lines

  query(statement, db_name, storage_name, field_name).first > 0
end

def storage_exists?(storage_name)
  statement = "SELECT COUNT(*)\nFROM `information_schema`.`tables`\nWHERE `table_type` = 'BASE TABLE'\nAND `table_schema` = ?\nAND `table_name` = ?\n".compress_lines

  query(statement, db_name, storage_name).first > 0
end
