def destroy_model_storage(repository, model)
  return true unless storage_exists?(model.storage_name(repository.name))
  success = without_notices { super }
  model.properties(repository.name).each do |property|
    drop_sequence(repository, property) if property.serial?
  end
  success
end

def field_exists?(storage_name, column_name)
  statement = "SELECT COUNT(*)\nFROM \"information_schema\".\"columns\"\nWHERE \"table_schema\" = current_schema()\nAND \"table_name\" = ?\nAND \"column_name\" = ?\n".compress_lines

  query(statement, storage_name, column_name).first > 0
end

def storage_exists?(storage_name)
  statement = "SELECT COUNT(*)\nFROM \"information_schema\".\"tables\"\nWHERE \"table_type\" = 'BASE TABLE'\nAND \"table_schema\" = current_schema()\nAND \"table_name\" = ?\n".compress_lines

  query(statement, storage_name).first > 0
end
