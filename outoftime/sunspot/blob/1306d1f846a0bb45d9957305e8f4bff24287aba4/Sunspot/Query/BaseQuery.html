def to_params
  params = {}
  if @keywords
    params[:q] = @keywords
    params[:fl] = '* score'
    params[:fq] = types_phrase
    params[:qf] = text_field_names.join(' ')
    params[:defType] = 'dismax'
  else
    params[:q] = types_phrase
  end
  params
end
