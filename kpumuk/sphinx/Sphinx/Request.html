def put_float(*floats)
  floats.each do |f|
    t1 = [f].pack('f') # machine order
    t2 = t1.unpack('L*').first # int in machine order
    @request << [t2].pack('N')
  end
end

def put_int64(*ints)
  ints.each { |i| @request << [i].pack('q').reverse }#[i >> 32, i & ((1 << 32) - 1)].pack('NN') }
end
