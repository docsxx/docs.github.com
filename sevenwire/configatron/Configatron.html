def temp(options = nil)

    temp_start(options)
    yield
  rescue Exception => e
    raise e
  ensure
    temp_end
  end

def temp_start(options = nil)
  n_space = rand
  @_store[n_space] = @_store[@_namespace.last].deep_clone
  @_namespace << n_space
  if options
    self.method_missing(:configure_from_hash, options)
  end
end
