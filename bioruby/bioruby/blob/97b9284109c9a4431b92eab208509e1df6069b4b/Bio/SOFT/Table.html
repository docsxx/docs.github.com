def add_header( line )
  raise "Can only define one header" unless @header.empty?      
  @header = @header.concat( parse_row( line ) )  # beware of clobbering this into an Array
  @header.each_with_index do |key, i|
    @header.column_index[key.downcase.to_sym] = i
  end
end
