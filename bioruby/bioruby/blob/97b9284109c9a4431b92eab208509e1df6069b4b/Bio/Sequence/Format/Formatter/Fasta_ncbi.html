def method_missing(sym, *args, &block) #:nodoc:
  begin
    @sequence.__send__(sym, *args, &block)
  rescue NoMethodError => evar
    lineno = __LINE__ - 2
    file = __FILE__
    bt_here = [ "#{file}:#{lineno}:in \`__send__\'",
                "#{file}:#{lineno}:in \`method_missing\'"
              ]
    if bt_here == evar.backtrace[0, 2] then
      bt = evar.backtrace[2..-1]
      evar = evar.class.new("undefined method \`#{sym.to_s}\' for #{self.inspect}")
      evar.set_backtrace(bt)
    end
    raise(evar)
  end
end

def output
  width = 70
  seq = @sequence.seq
  #gi = @sequence.gi_number
  dbname = 'lcl'
  if @sequence.primary_accession.to_s.empty? then
    idstr = @sequence.entry_id
  else
    idstr = "#{@sequence.primary_accession}.#{@sequence.sequence_version}"
  end

  definition = @sequence.definition
  header = "#{dbname}|#{idstr} #{definition}"

  ">#{header}\n" + seq.to_s.gsub(Regexp.new(".{1,#{width}}"), "\\0\n")
end
