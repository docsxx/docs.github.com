def composition
  unless @data['COMPOSITION']
    hash = Hash.new(0)
    fetch('COMPOSITION').scan(/\((\S+)\)(\d+)/).each do |key, val|
      hash[key] = val.to_i
    end
    @data['COMPOSITION'] = hash
  end
  @data['COMPOSITION']
end

def enzymes
  unless @data['ENZYME']
    field = fetch('ENZYME')
    if /\(/.match(field)  # old version
      @data['ENZYME'] = field.scan(/\S+ \(\S+\)/)
    else
      @data['ENZYME'] = field.scan(/\S+/)
    end
  end
  @data['ENZYME']
end

def references
  unless @data['REFERENCE']
    ary = Array.new
    lines = lines_fetch('REFERENCE')
    lines.each do |line|
      if /^\d+\s+\[PMID/.match(line)
        ary << line
      else
        ary.last << " #{line.strip}"
      end
    end
    @data['REFERENCE'] = ary
  end
  @data['REFERENCE']
end
