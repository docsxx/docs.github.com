def dblinks_as_hash
  hash = {}
  dblinks.each do |line|
    name, *list = line.split(/\s+/)
    db = name.downcase.sub(/:/, '')
    hash[db] = list
  end
  return hash
end

def genes_as_hash
  hash = {}
  genes.each do |line|
    name, *list = line.split(/\s+/)
    org = name.downcase.sub(/:/, '')
    genes = list.map {|x| x.sub(/\(.*\)/, '')}
    #names = list.map {|x| x.scan(/.*\((.*)\)/)}
    hash[org] = genes
  end
  return hash
end
