def percent_identity
  unless defined?(@percent_identity)
    x = field_fetch('overall percent identity', @d0)
    @percent_identity = 
      (/([\d\.]+)\s*\%/ =~ x.to_s) ? $1 : nil
  end
  @percent_identity
end
