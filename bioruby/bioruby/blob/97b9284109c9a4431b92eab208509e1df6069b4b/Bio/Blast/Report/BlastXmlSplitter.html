def get_entry
  if @parsed_entries.empty? then
    @raw_unsupported = false
    ent = super
    prepare_parsed_entries(ent)
    self.parsed_entry = @parsed_entries.shift
  else
    raise 'not supported for new BLAST XML format'
  end
  ent
end

def get_parsed_entry
  if @parsed_entries.empty? then
    ent = get_entry
  else
    self.parsed_entry = @parsed_entries.shift
    self.entry = nil
    @raw_unsupported = true
  end
  self.parsed_entry
end
