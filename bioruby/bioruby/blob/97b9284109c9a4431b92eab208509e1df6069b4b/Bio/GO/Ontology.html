def self.parse_goids(line)
  goids = []
  loop {
    if /^ *[$%<]\S.+?;/ =~ line
      endpoint = line.index(';') + 1
      line = line[endpoint..line.size]
    elsif /^,* GO:(\d{7}),*/ =~ line
      goids << $1.clone
      endpoint = line.index(goids.last) + goids.last.size
      line = line[endpoint..line.size]
    else
      break
    end
  }
  return goids
end

def goid2term(goid)
  term = id2term[goid]
  term = id2term[id2id[goid]] if term == nil
  return term
end
