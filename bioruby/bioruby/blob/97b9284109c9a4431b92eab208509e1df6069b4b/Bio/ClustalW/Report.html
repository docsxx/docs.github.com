def initialize(str, seqclass = nil)
  @raw = str
  @align = nil
  @match_line = nil
  @header = nil
  case seqclass
  when /PROTEIN/i
    @seqclass = Bio::Sequence::AA
  when /[DR]NA/i
    @seqclass = Bio::Sequence::NA
  else
    if seqclass.is_a?(Module) then
      @seqclass = seqclass
    else
      @seqclass = Bio::Sequence
    end
  end
end
