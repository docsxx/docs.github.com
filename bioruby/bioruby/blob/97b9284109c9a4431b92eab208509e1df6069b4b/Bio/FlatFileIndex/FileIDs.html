def [](n)
  r = super(n)
  if r then
    r
  else
    data = @hash["#{@prefix}#{n}"]
    if data then
      self[n] = data
    end
    super(n)
  end
end

def []=(n, data)
  if data.is_a?(FileID) then
    super(n, data)
  elsif data then
    super(n, FileID.new_from_string(data))
  else
    # data is nil
    super(n, nil)
  end
  self[n]
end

def cache_all
  a = @hash.keys.collect do |k|
    if k =~ /\A#{Regexp.escape(@prefix)}(\d+)/ then
      $1.to_i
    else
      nil
    end
  end
  a.compact!
  a.each do |i|
    self[i]
  end
  a
end
