Class: Bio::Blat::Report::SeqDesc

def initialize(gap_count, gap_bases, name, size,
               st, ed, starts, seqs)
  @gap_count = gap_count.to_i
  @gap_bases = gap_bases.to_i
  @name = name
  @size = size.to_i
  @start = st.to_i
  @end = ed.to_i
  @starts = starts.collect { |x| x.to_i }
  @seqs = seqs
end
