def initialize(data)
  @definition, *data = data.split(/\n/)
  @data = {}
  @sequence = ''

  pat = /;\s+([^:]+):\s+(.*)/

  data.each do |x|
    if pat.match(x)
      @data[$1] = $2
    else
      @sequence += x
    end
  end
end
