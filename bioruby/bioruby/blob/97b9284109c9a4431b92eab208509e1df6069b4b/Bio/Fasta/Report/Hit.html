def initialize(data)
  score, query, target = data.split(/\n>/)

  @definition, *score = score.split(/\n/)
  @score = {}

  pat = /;\s+([^:]+):\s+(.*)/

  score.each do |x|
    if pat.match(x)
      @score[$1] = $2
    end
  end

  @query = Query.new(query)
  @target = Target.new(target)
end

def bit_score
  if @score['fa_bits']
    @score['fa_bits'].to_f
  elsif @score['sw_bits']
    @score['sw_bits'].to_f
  elsif @score['fx_bits']
    @score['fx_bits'].to_f
  elsif @score['tx_bits']
    @score['tx_bits'].to_f
  end
end
