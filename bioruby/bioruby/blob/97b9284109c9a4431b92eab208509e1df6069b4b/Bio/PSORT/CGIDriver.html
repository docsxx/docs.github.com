def initialize(host = '', path = '')
  case host.to_s
  when /^http:/
    uri = host.to_s
  else
    uri = 'http://' + host + '/' + path
  end
  @uri = URI.parse(uri)
  @args = {}
  @report = ''
end

def exec(query)
  data = make_args(query)  

  begin
    result = nil
    Bio::Command.start_http(@uri.host) {|http|
      result = http.post(@uri.path, data)
    }
    @report = result.body
    output = parse_report(@report)
  end

  return output
end
