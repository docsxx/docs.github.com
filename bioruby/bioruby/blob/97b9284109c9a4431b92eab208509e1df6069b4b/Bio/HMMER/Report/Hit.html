def initialize(hit_data)
  @hsps = Array.new
  if /^(\S+)\s+(.*?)\s+(\S+)\s+(\S+)\s+(\S+)$/ =~ hit_data
    @accession, @description, @score, @evalue, @num = \
    [$1, $2, $3.to_f, $4.to_f, $5.to_i]
  end
end

def target_def
  if @hsps.size == 1
    "<#{@hsps[0].domain}> #{@description}"
  else
    "<#{@num.to_s}> #{@description}"
  end
end
