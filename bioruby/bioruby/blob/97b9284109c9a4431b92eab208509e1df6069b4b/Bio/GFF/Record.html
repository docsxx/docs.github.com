def initialize(str)
  @comment = str.chomp[/#.*/]
  return if /^#/.match(str)
  @seqname, @source, @feature, @start, @end, @score, @strand, @frame,
    attributes, = str.chomp.split("\t")
  @attributes = parse_attributes(attributes) if attributes
end
