def append_suffixes(array, text, offset, base_offset, index)
  sc = StringScanner.new(text)
  sc.skip(/[^A-Za-z_]+/)
  until sc.eos?
    array << [(sc.pos + offset), base_offset, index]
    break unless sc.skip(/[A-Za-z_][A-Za-z0-9_]*[^A-Za-z_]*/)
  end
  array
end
