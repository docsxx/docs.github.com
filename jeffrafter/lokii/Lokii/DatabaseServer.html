def check
  messages = Lokii::Inbox.pending.find(:all)
  messages.each {|message|
    handle({:phone => message.phone,
            :number => message.number,
            :text => message.text,
            :created_at => message.insertdate,
            :processed_at => Time.now})
  }      
end
