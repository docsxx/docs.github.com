def delete( arg )
  case arg
    when @the_many_model
      result = @the_many_model.dbh.do(
        %{
          DELETE FROM #{@the_many_model.table}
          WHERE
            #{@the_one_fk} = ?
            AND #{@the_many_model.pk_clause}
        },
        @the_one.pk,
        arg.pk
      )
      result > 0
    when Hash
      hash = arg
      keys = hash.keys
      where_subclause = keys.map { |k|
        "#{k} = ?"
      }.join( " AND " )
      @the_many_model.dbh.do(
        %{
          DELETE FROM #{@the_many_model.table}
          WHERE
            #{@the_one_fk} = ?
            AND #{where_subclause}
        },
        @the_one.pk,
        *( keys.map { |k| hash[ k ] } )
      )
  end
end
