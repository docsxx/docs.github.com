def ancestral_trait
  if respond_to?(:ancestors)
    ancs = ancestors
  else
    ancs = self.class.ancestors
  end
  ancs.reverse.inject({}){|s,v| s.merge(v.trait)}.merge(trait)
end

def ancestral_trait_class_reader( *names )

def ancestral_trait_reader( *names )

  end

trait

public trait(hash = nil)

Adds a method to Object to annotate your objects with certain traits. It’s basically a simple Hash that takes the current object as key

Example: 

  class Foo
    trait :instance => false

    def initialize
      trait :instance => true
    end
  end

  Foo.trait[:instance]
  # false

  foo = Foo.new
  foo.trait[:instance]
  # true

def trait hash = nil
  if hash
    Traits[self].merge! hash
  else
    Traits[self]
  end
end
