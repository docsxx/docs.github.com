def initialize( rows, cols, top_row, left_col, buffer_initial = nil )
  @rows, @cols = rows, cols
  @top_row, @left_col = top_row, left_col
  @views = Hash.new { |h,k| h[ k ] = View.new }
  @buffer_current = buffer_initial
  @default_formatting = $diakonos.settings[ "lang.#{@language}.format.default" ] || Curses::A_NORMAL

  @win_main         = nil
  @win_context      = nil
  @win_status       = nil
  @win_interaction  = nil
  @win_line_numbers = nil

  @pane_child_dir   = nil
  @pane_child       = nil

  @iline_mutex      = Mutex.new
end

def display
  return  if $diakonos.testing
  return  if ! $diakonos.do_display

  Thread.new do
    if ! $diakonos.display_mutex.try_lock
      $diakonos.display_enqueue self
    else
      begin
        Curses::curs_set 0
        @continued_format_class = nil
        @pen_down = true

        # First, we have to "draw" off-screen, in order to check for opening of
        # multi-line highlights.

        # So, first look backwards from the @buffer_current.top_line to
        # find the first opening regexp match, if any.
        b = @buffer_current
        v = @views[ b ]
        index = v.top_line - 1
        b[
          [
            0,
            v.top_line - $diakonos.settings[ "view.lookback" ]
          ].max...v.top_line
        ].reverse_each do |line|
          open_index = -1
          open_token_class = nil
          open_match_text = nil

          open_index, open_token_class, open_match_text = find_opening_match( line )

          if open_token_class
            @pen_down = false
            b[ index...v.top_line ].each do |line|
              print_line line
            end
            @pen_down = true

            break
          end

          index = index - 1
        end

        # Draw each on-screen line.
        y = 0
        b[ v.top_line...($diakonos.win_main_height + v.top_line) ].each_with_index do |line, row|
          if @win_line_numbers
            @win_line_numbers.setpos( y, 0 )
            @win_line_numbers.attrset $diakonos.settings[ 'view.line_numbers.format' ]
            n = ( v.top_line+row+1 ).to_s
            @win_line_numbers.addstr(
              $diakonos.settings[ 'view.line_numbers.number_format' ] % [
                n[ -[ $diakonos.settings[ 'view.line_numbers.width' ], n.length ].min..-1 ]
              ]
            )
          end
          @win_main.setpos( y, 0 )
          print_line line.expand_tabs( @tab_size )
          @win_main.setpos( y, 0 )
          paint_marks v.top_line + row
          y += 1
        end

        # Paint the empty space below the file if the file is too short to fit in one screen.
        ( y...$diakonos.win_main_height ).each do |y|
          if @win_line_numbers
            @win_line_numbers.setpos( y, 0 )
            @win_line_numbers.attrset $diakonos.settings[ 'view.line_numbers.format' ]
            @win_line_numbers.addstr( ' ' * $diakonos.settings[ 'view.line_numbers.width' ] )
          end

          @win_main.setpos( y, 0 )
          @win_main.attrset @default_formatting
          linestr = " " * win_main_width
          if $diakonos.settings[ "view.nonfilelines.visible" ]
            linestr[ 0 ] = $diakonos.settings[ "view.nonfilelines.character" ] || "~"
          end

          @win_main.addstr linestr
        end

        paint_column_markers

        if @win_line_numbers
          @win_line_numbers.refresh
        end
        @win_main.setpos( v.y , v.x )
        @win_main.refresh

        # if @language != @original_language
          # set_language @original_language
        # end

        Curses::curs_set 1
      rescue Exception => e
        $diakonos.log( "Display Exception:" )
        $diakonos.log( e.message )
        $diakonos.log( e.backtrace.join( "\n" ) )
        show_exception e
      end

      $diakonos.display_mutex.unlock
      $diakonos.display_dequeue
    end

  end

end

def setup
  if $diakonos.settings[ 'view.line_numbers' ]
    @win_main = ::Diakonos::Window.new(
      win_main_height,
      win_main_width,
      @top_row,
      @left_col + $diakonos.settings[ 'view.line_numbers.width' ]
    )
    @win_line_numbers = ::Diakonos::Window.new(
      win_main_height,
      $diakonos.settings[ 'view.line_numbers.width' ],
      @top_row,
      @left_col
    )
  else
    @win_main = ::Diakonos::Window.new( win_main_height, @cols, @top_row, @left_col )
    @win_line_numbers = nil
  end
  @win_status = ::Diakonos::Window.new( 1, @cols, @top_row + @rows - 2, @left_col )
  @win_status.attrset $diakonos.settings[ 'status.format' ]
  @win_interaction = ::Diakonos::Window.new( 1, @cols, @top_row + @rows - 1, @left_col )

  if $diakonos.settings[ 'context.visible' ]
    pos = $diakonos.settings[ 'context.combined' ] ? 1 : 3
    @win_context = ::Diakonos::Window.new( 1, @cols, @top_row + @rows - pos, @left_col )
  else
    @win_context = nil
  end

  if ! $diakonos.testing
    @win_main.keypad true
    @win_status.keypad true
    @win_interaction.keypad true
    if @win_line_numbers
      @win_line_numbers.keypad true
    end
    if @win_context
      @win_context.keypad true
    end
  end

  @win_interaction.refresh
  @win_main.refresh
  if @win_line_numbers
    @win_line_numbers.refresh
  end

  if @pane_child
    @pane_child.setup
  end
end

def update_context_line( context )
  Curses::curs_set 0
  @win_context.setpos( 0, 0 )
  chars_printed = 0

  if context.length > 0
    truncation = [ $diakonos.settings[ "context.max_levels" ], context.length ].min
    max_length = [
      ( @cols / truncation ) - $diakonos.settings[ "context.separator" ].length,
      ( $diakonos.settings[ "context.max_segment_width" ] || @cols )
    ].min
    line = nil
    context_subset = context[ 0...truncation ]
    context_subset = context_subset.collect do |line|
      line.strip[ 0...max_length ]
    end

    context_subset.each do |line|
      @win_context.attrset $diakonos.settings[ "context.format" ]
      @win_context.addstr line
      chars_printed += line.length
      @win_context.attrset $diakonos.settings[ "context.separator.format" ]
      @win_context.addstr $diakonos.settings[ "context.separator" ]
      chars_printed += $diakonos.settings[ "context.separator" ].length
    end
  end

  @iline_mutex.synchronize do
    @win_context.attrset $diakonos.settings[ "context.format" ]
    @win_context.addstr( " " * ( @cols - chars_printed ) )
    @win_context.refresh
  end

  $diakonos.display_mutex.synchronize do
    if @buffer_current
      y, x = @buffer_current.last_screen_y, @buffer_current.last_screen_x
    else
      y, x = 0,0
    end
    position_cursor_main y, x
    @win_main.refresh
  end

  Curses::curs_set 1
end

def win_main_height
  # One line for the status line
  # One line for the input line
  h = @rows - 2
  # One line for the context line
  if $diakonos.settings[ "context.visible" ] && ! $diakonos.settings[ "context.combined" ]
    h -= 1
  end
  h
end
