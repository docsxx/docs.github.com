Class: Diakonos::Pane

def height_main
  # One line for the status line
  # One line for the input line
  h = @rows - 2
  # One line for the context line
  if $diakonos.settings[ "context.visible" ] && ! $diakonos.settings[ "context.combined" ]
    h -= 1
  end
  h
end

def setup
  if $diakonos.settings[ 'view.line_numbers' ]
    @win_main = ::Diakonos::Window.new(
      height_main,
      @cols - $diakonos.settings[ 'view.line_numbers.width' ],
      @top_row,
      @left_col + $diakonos.settings[ 'view.line_numbers.width' ]
    )
    @win_line_numbers = ::Diakonos::Window.new(
      height_main,
      $diakonos.settings[ 'view.line_numbers.width' ],
      @top_row,
      @left_col
    )
  else
    @win_main = ::Diakonos::Window.new( height_main, @cols, @top_row, @left_col )
    @win_line_numbers = nil
  end
  @win_status = ::Diakonos::Window.new( 1, @cols, @top_row + @rows - 2, @left_col )
  @win_status.attrset $diakonos.settings[ 'status.format' ]
  @win_interaction = ::Diakonos::Window.new( 1, @cols, @top_row + @rows - 1, @left_col )

  if $diakonos.settings[ 'context.visible' ]
    pos = $diakonos.settings[ 'context.combined' ] ? 1 : 3
    @win_context = ::Diakonos::Window.new( 1, @cols, @top_row + @rows - pos, @left_col )
  else
    @win_context = nil
  end

  if ! $diakonos.testing
    @win_main.keypad true
    @win_status.keypad true
    @win_interaction.keypad true
    if @win_line_numbers
      @win_line_numbers.keypad true
    end
    if @win_context
      @win_context.keypad true
    end
  end

  @win_interaction.refresh
  @win_main.refresh
  if @win_line_numbers
    @win_line_numbers.refresh
  end

  if @child_pane
    @child_pane.setup
  end
end

def update_context_line( context )
  Curses::curs_set 0
  @win_context.setpos( 0, 0 )
  chars_printed = 0

  if context.length > 0
    truncation = [ $diakonos.settings[ "context.max_levels" ], context.length ].min
    max_length = [
      ( @cols / truncation ) - $diakonos.settings[ "context.separator" ].length,
      ( $diakonos.settings[ "context.max_segment_width" ] || @cols )
    ].min
    line = nil
    context_subset = context[ 0...truncation ]
    context_subset = context_subset.collect do |line|
      line.strip[ 0...max_length ]
    end

    context_subset.each do |line|
      @win_context.attrset $diakonos.settings[ "context.format" ]
      @win_context.addstr line
      chars_printed += line.length
      @win_context.attrset $diakonos.settings[ "context.separator.format" ]
      @win_context.addstr $diakonos.settings[ "context.separator" ]
      chars_printed += $diakonos.settings[ "context.separator" ].length
    end
  end

  @iline_mutex.synchronize do
    @win_context.attrset $diakonos.settings[ "context.format" ]
    @win_context.addstr( " " * ( @cols - chars_printed ) )
    @win_context.refresh
  end

  $diakonos.display_mutex.synchronize do
    @win_main.setpos( @current_buffer.last_screen_y, @current_buffer.last_screen_x )
    @win_main.refresh
  end

  Curses::curs_set 1
end
