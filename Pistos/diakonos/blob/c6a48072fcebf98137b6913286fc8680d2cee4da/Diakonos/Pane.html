def cursor_to( row, col, do_display = DONT_DISPLAY, stopped_typing = STOPPED_TYPING, adjust_row = ADJUST_ROW )
  buffer, view = b_v
  old_row, old_col = view.row, view.col

  row = row.fit( 0, buffer.length - 1 )

  if col < 0
    if ! adjust_row
      col = 0
    else
      if row <= 0
        col = 0
      else
        row = row - 1
        col = buffer[ row ].length
      end
    end
  elsif col > buffer[ row ].length
    if ! adjust_row
      col = buffer[ row ].length
    else
      if row >= buffer.length - 1
        col = buffer[ row ].length
      else
        row = row + 1
        col = 0
      end
    end
  end

  if adjust_row
    view.desired_column = col
  else
    goto_col = [ view.desired_column, buffer[ row ].length ].min
    if col < goto_col
      col = goto_col
    end
  end

  new_col = buffer.tab_expanded_column( col, row )
  view_changed = show_character( row, new_col )
  screen_row = row - view.top_line
  screen_col = new_col - view.left_col

  if stopped_typing
    buffer.typing = false
  end
  view.row = row
  view.col = col

  changed = ( row != old_row || col != old_col )
  if changed
    buffer.record_mark_start_and_end

    removed = false
    if ! buffer.changing_selection && buffer.selecting?
      buffer.remove_selection DONT_DISPLAY
      removed = true
    end

    highlight_changed = @buffer_current.update_highlight( row, col )

    if removed || ( do_display && ( buffer.selecting? || view_changed || highlight_changed ) )
      display
    else
      $diakonos.display_mutex.synchronize do
        @win_main.setpos( screen_row, screen_col )
      end
    end
    $diakonos.update_status_line
    $diakonos.update_context_line
  end

  changed
end

def cursor_to_bol
  row = view_current.row

  col = case $diakonos.settings[ "bol_behaviour" ]
  when BOL_ZERO
    0
  when BOL_FIRST_CHAR
    ( @buffer_current[ row ] =~ /\S/ ) || 0
  when BOL_ALT_ZERO
    view_current.col != 0 ? 0 : @buffer_current[ row ] =~ /\S/
  #when BOL_ALT_FIRST_CHAR
  else
    first_char_col = ( ( @buffer_current[ row ] =~ /\S/ ) || 0 )
    view_current.col == first_char_col ?  0 : first_char_col
  end

  cursor_to( row, col, DO_DISPLAY )
end

def cursor_to_eol
  y = @win_main.cury
  end_col = @buffer_current.line_at( y ).length
  last_char_col = @buffer_current.line_at( y ).rstrip.length

  col = case $diakonos.settings[ 'eol_behaviour' ]
  when EOL_END
    end_col
  when EOL_LAST_CHAR
    last_char_col
  when EOL_ALT_LAST_CHAR
    view_current.col == last_char_col ? end_col : last_char_col
  else
    view_current.col == end_col ? last_char_col : end_col
  end

  cursor_to( view_current.row, col, DO_DISPLAY )
end

def display
  return  if $diakonos.testing
  return  if ! $diakonos.do_display

  Thread.new do
    if ! $diakonos.display_mutex.try_lock
      $diakonos.display_enqueue self
    else
      begin
        Curses::curs_set 0
        @continued_format_class = nil
        @pen_down = true

        # First, we have to "draw" off-screen, in order to check for opening of
        # multi-line highlights.

        # So, first look backwards from the @buffer_current.top_line to
        # find the first opening regexp match, if any.
        b, v = b_v
        index = v.top_line - 1
        b[
          [
            0,
            v.top_line - $diakonos.settings[ "view.lookback" ]
          ].max...v.top_line
        ].reverse_each do |line|
          open_index = -1
          open_token_class = nil
          open_match_text = nil

          open_index, open_token_class, open_match_text = b.find_opening_match( line )

          if open_token_class
            @pen_down = false
            b[ index...v.top_line ].each do |line|
              print_line line
            end
            @pen_down = true

            break
          end

          index = index - 1
        end

        # Draw each on-screen line.
        y = 0
        b[ v.top_line...(win_main_height + v.top_line) ].each_with_index do |line, row|
          if @win_line_numbers
            @win_line_numbers.setpos( y, 0 )
            @win_line_numbers.attrset $diakonos.settings[ 'view.line_numbers.format' ]
            n = ( v.top_line+row+1 ).to_s
            @win_line_numbers.addstr(
              $diakonos.settings[ 'view.line_numbers.number_format' ] % [
                n[ -[ $diakonos.settings[ 'view.line_numbers.width' ], n.length ].min..-1 ]
              ]
            )
          end
          @win_main.setpos( y, 0 )
          print_line line.expand_tabs( b.tab_size )
          @win_main.setpos( y, 0 )
          paint_marks v.top_line + row
          y += 1
        end

        # Paint the empty space below the file if the file is too short to fit in one screen.
        ( y...win_main_height ).each do |y|
          if @win_line_numbers
            @win_line_numbers.setpos( y, 0 )
            @win_line_numbers.attrset $diakonos.settings[ 'view.line_numbers.format' ]
            @win_line_numbers.addstr( ' ' * $diakonos.settings[ 'view.line_numbers.width' ] )
          end

          @win_main.setpos( y, 0 )
          @win_main.attrset @default_formatting
          linestr = " " * win_main_width
          if $diakonos.settings[ "view.nonfilelines.visible" ]
            linestr[ 0 ] = $diakonos.settings[ "view.nonfilelines.character" ] || "~"
          end

          @win_main.addstr linestr
        end

        paint_column_markers

        if @win_line_numbers
          @win_line_numbers.refresh
        end
        @win_main.setpos( v.row - v.top_line , v.col - v.left_col )
        @win_main.refresh

        # if @language != @original_language
          # set_language @original_language
        # end

        Curses::curs_set 1
      rescue Exception => e
        $diakonos.log( "Display Exception:" )
        $diakonos.log( e.message )
        $diakonos.log( e.backtrace.join( "\n" ) )
        show_exception e
      end

      $diakonos.display_mutex.unlock
      $diakonos.display_dequeue
    end

  end

end

def paint_column_markers
  $diakonos.column_markers.each_value do |data|
    column = data[ :column ]
    next  if column.nil?
    next  if column > win_main_width - view_current.left_col || column - view_current.left_col < 0

    num_lines_to_paint = [ win_main_height, @buffer_current.size - view_current.top_line ].min
    ( 0...num_lines_to_paint ).each do |row|
      @win_main.setpos( row, column - view_current.left_col )
      @win_main.attrset data[ :format ]
      @win_main.addstr( @buffer_current[ view_current.top_line + row ][ column + view_current.left_col ] || ' ' )
    end
  end
end

def paint_marks( row )
  string = @buffer_current[ row ][ view_current.left_col...view_current.left_col + win_main_width ]
  return  if string.nil? || string.empty?

  string = string.expand_tabs( @buffer_current.tab_size )
  cury = @win_main.cury
  curx = @win_main.curx

  @buffer_current.text_marks.values.flatten.reverse_each do |text_mark|
    next  if text_mark.nil?

    @win_main.attrset text_mark.formatting

    case @selection_mode
    when :normal
      if ( (text_mark.start_row + 1) .. (text_mark.end_row - 1) ) === row
        @win_main.setpos( cury, curx )
        @win_main.addstr string
      elsif row == text_mark.start_row and row == text_mark.end_row
        paint_single_row_mark( row, text_mark, string, curx, cury )
      elsif row == text_mark.start_row
        expanded_col = @buffer_current.tab_expanded_column( text_mark.start_col, row )
        if expanded_col < view_current.left_col + win_main_width
          left = [ expanded_col - view_current.left_col, 0 ].max
          @win_main.setpos( cury, curx + left )
          @win_main.addstr string[ left..-1 ]
        end
      elsif row == text_mark.end_row
        right = @buffer_current.tab_expanded_column( text_mark.end_col, row ) - view_current.left_col
        @win_main.setpos( cury, curx )
        @win_main.addstr string[ 0...right ]
      else
        # This row not in selection.
      end
    when :block
      if(
        text_mark.start_row <= row && row <= text_mark.end_row ||
        text_mark.end_row <= row && row <= text_mark.start_row
      )
        paint_single_row_mark( row, text_mark, string, curx, cury )
      end
    end
  end
end

def paint_single_row_mark( row, text_mark, string, curx, cury )
  expanded_col = @buffer_current.tab_expanded_column( text_mark.start_col, row )
  if expanded_col < view_current.left_col + win_main_width
    left = [ expanded_col - view_current.left_col, 0 ].max
    right = @buffer_current.tab_expanded_column( text_mark.end_col, row ) - view_current.left_col
    if left < right
      @win_main.setpos( cury, curx + left )
      @win_main.addstr string[ left...right ]
    end
  end
end

def pitch_view_to( new_top_line, do_pitch_cursor = DONT_PITCH_CURSOR, do_display = DO_DISPLAY )
  old_top_line = view_current.top_line

  if new_top_line < 0
    view_current.top_line = 0
  elsif new_top_line + win_main_height > @buffer_current.length
    view_current.top_line = [ @buffer_current.length - win_main_height, 0 ].max
  else
    view_current.top_line = new_top_line
  end

  top_line = view_current.top_line

  old_row = view_current.row
  old_col = view_current.col

  changed = ( top_line - old_top_line )
  if changed != 0 && do_pitch_cursor
    view_current.row += changed
  end

  height = [ win_main_height, @buffer_current.length ].min

  new_row = view_current.row.fit( top_line, top_line + height - 1 )
  if new_row - top_line < $diakonos.settings[ "view.margin.y" ]
    new_row = top_line + $diakonos.settings[ "view.margin.y" ]
    new_row = new_row.fit( top_line, top_line + height - 1 )
  elsif top_line + height - 1 - new_row < $diakonos.settings[ "view.margin.y" ]
    new_row = top_line + height - 1 - $diakonos.settings[ "view.margin.y" ]
    new_row = new_row.fit( top_line, top_line + height - 1 )
  end
  view_current.row = new_row

  view_current.col = view_current.col.fit(
    view_current.left_col,
    [
      view_current.left_col + win_main_width - 1,
      @buffer_current[ view_current.row ].length
    ].min
  )
  # @last_screen_y = view_current.row - top_line
  # @last_screen_x = tab_expanded_column( view_current.col, view_current.row ) - view_current.left_col

  @buffer_current.record_mark_start_and_end

  if changed != 0
    if ! @buffer_current.changing_selection && @buffer_current.selecting?
      @buffer_current.remove_selection DONT_DISPLAY
    end

    @buffer_current.highlight_matches
    if $diakonos.there_was_non_movement
      $diakonos.push_cursor_state( old_top_line, old_row, old_col )
    end
  end

  if do_display
    # $diakonos.display_buffer @buffer_current
    display
  end

  changed
end

def print_line( line )
  i = 0
  substr = nil
  index = nil
  while i < line.length
    substr = line[ i..-1 ]
    if @continued_format_class
      close_index, close_match_text = @buffer_current.find_closing_match(
        substr,
        @buffer_current.close_token_regexps[ @continued_format_class ],
        i == 0
      )

      if close_match_text.nil?
        print_string truncate_off_screen( substr, i )
        print_padding_from( line.length )
        i = line.length
      else
        end_index = close_index + close_match_text.length
        print_string truncate_off_screen( substr[ 0...end_index ], i )
        @continued_format_class = nil
        i += end_index
      end
    else
      first_index, first_token_class, first_word = @buffer_current.find_opening_match( substr, MATCH_ANY, i == 0 )

      if @lang_stack.length > 0
        prev_lang, close_token_class = @lang_stack[ -1 ]
        close_index, close_match_text = @buffer_current.find_closing_match(
          substr,
          $diakonos.close_token_regexps[ prev_lang ][ close_token_class ],
          i == 0
        )
        if close_match_text && close_index <= first_index
          if close_index > 0
            # Print any remaining text in the embedded language
            print_string truncate_off_screen( substr[ 0...close_index ], i )
            i += substr[ 0...close_index ].length
          end

          @lang_stack.pop
          set_language prev_lang

          print_string(
            truncate_off_screen( substr[ close_index...(close_index + close_match_text.length) ], i ),
            @buffer_current.token_formats[ close_token_class ]
          )
          i += close_match_text.length

          # Continue printing from here.
          next
        end
      end

      if first_word
        if first_index > 0
          # Print any preceding text in the default format
          print_string truncate_off_screen( substr[ 0...first_index ], i )
          i += substr[ 0...first_index ].length
        end
        print_string(
          truncate_off_screen( first_word, i ),
          @buffer_current.token_formats[ first_token_class ]
        )
        i += first_word.length
        if @buffer_current.close_token_regexps[ first_token_class ]
          change_to = $diakonos.settings[ "lang.#{@buffer_current.language}.tokens.#{first_token_class}.change_to" ]
          if change_to
            @lang_stack.push [ @buffer_current.language, first_token_class ]
            set_language change_to
          else
            @continued_format_class = first_token_class
          end
        end
      else
        print_string truncate_off_screen( substr, i )
        i += substr.length
        break
      end
    end
  end

  print_padding_from i
end

def show_character( row, col )
  v = view_current
  old_top_line = v.top_line
  old_left_col = v.left_col

  while row < v.top_line + $diakonos.settings[ "view.margin.y" ]
    amount = (-1) * $diakonos.settings[ "view.jump.y" ]
    break  if( pitch_view( amount, DONT_PITCH_CURSOR, DONT_DISPLAY ) != amount )
  end
  while row > v.top_line + win_main_height - 1 - $diakonos.settings[ "view.margin.y" ]
    amount = $diakonos.settings[ "view.jump.y" ]
    break  if( pitch_view( amount, DONT_PITCH_CURSOR, DONT_DISPLAY ) != amount )
  end

  while col < v.left_col + $diakonos.settings[ "view.margin.x" ]
    amount = (-1) * $diakonos.settings[ "view.jump.x" ]
    break  if( pan_view( amount, DONT_DISPLAY ) != amount )
  end
  while col > v.left_col + win_main_width - $diakonos.settings[ "view.margin.x" ] - 2
    amount = $diakonos.settings[ "view.jump.x" ]
    break  if( pan_view( amount, DONT_DISPLAY ) != amount )
  end

  v.top_line != old_top_line || v.left_col != old_left_col
end

def truncate_off_screen( string, write_cursor_col )
  retval = string

  # Truncate based on left edge of display area
  if write_cursor_col < view_current.left_col
    retval = retval[ (view_current.left_col - write_cursor_col)..-1 ]
    write_cursor_col = view_current.left_col
  end

  if retval
    # Truncate based on right edge of display area
    if write_cursor_col + retval.length > view_current.left_col + win_main_width - 1
      new_length = ( view_current.left_col + win_main_width - write_cursor_col )
      if new_length <= 0
        retval = nil
      else
        retval = retval[ 0...new_length ]
      end
    end
  end

  retval == "" ? nil : retval
end

def update_context_line( context )
  Curses::curs_set 0
  @win_context.setpos( 0, 0 )
  chars_printed = 0

  if context.length > 0
    truncation = [ $diakonos.settings[ "context.max_levels" ], context.length ].min
    max_length = [
      ( @cols / truncation ) - $diakonos.settings[ "context.separator" ].length,
      ( $diakonos.settings[ "context.max_segment_width" ] || @cols )
    ].min
    line = nil
    context_subset = context[ 0...truncation ]
    context_subset = context_subset.collect do |line|
      line.strip[ 0...max_length ]
    end

    context_subset.each do |line|
      @win_context.attrset $diakonos.settings[ "context.format" ]
      @win_context.addstr line
      chars_printed += line.length
      @win_context.attrset $diakonos.settings[ "context.separator.format" ]
      @win_context.addstr $diakonos.settings[ "context.separator" ]
      chars_printed += $diakonos.settings[ "context.separator" ].length
    end
  end

  @iline_mutex.synchronize do
    @win_context.attrset $diakonos.settings[ "context.format" ]
    @win_context.addstr( " " * ( @cols - chars_printed ) )
    @win_context.refresh
  end

  $diakonos.display_mutex.synchronize do
    if @buffer_current
      y, x = @buffer_current.last_screen_y, @buffer_current.last_screen_x
    else
      y, x = 0,0
    end
    position_cursor_main y, x
    @win_main.refresh
  end

  Curses::curs_set 1
end
