def initialize(path, fs_path)
  super(path) { File.open(fs_path, 'rb') }
  @fs_path = fs_path
  WebsiteAccess.website.cache[[:fs_path, @fs_path]] = File.mtime(@fs_path)
  @meta_info['modified_at'] = File.mtime(@fs_path)
end
