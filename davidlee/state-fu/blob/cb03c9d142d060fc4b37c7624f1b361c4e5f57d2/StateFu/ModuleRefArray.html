def inject_into( obj )
  metaclass = class << obj; self; end
  mods = self.modules()
  metaclass.class_eval do
    mods.each do |mod|
      include( mod )
    end
  end
end

def modules
  self.map do |h|
    case h
    when String, Symbol
      Object.const_get( h.to_s.classify )
    when Module
      h
    else
      raise ArgumentError.new( h.class.inspect )
    end
  end
end # modules
