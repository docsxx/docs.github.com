def cluster(name = nil, a = {}, &block)
  if name && name = "cluster_#{name}"
    subgraph( name, a, &block )
  else
    clusters
  end
end

def subgraph(name, a = {})
  returning( SubGraph.new(name, a)) do |g|
    @subgraphs << g
    yield g if block_given?
  end
end

def to_str
  build(["digraph #{quote name} {",
         [
          comment("global options"),
          "graph #{graph};",
          "node #{node};",
          "edge #{edge};"
         ],
         comment("nodes"),
         nodes.map(&:to_str),
         comment("links"),
         links.map(&:to_str),
         comment("subgraphs"),
         subgraphs.map(&:to_str),
         "}"])
end
