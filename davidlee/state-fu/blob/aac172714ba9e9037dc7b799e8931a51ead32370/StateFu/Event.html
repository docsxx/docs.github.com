def from *args
  options = args.extract_options!.symbolize_keys!
  args.flatten!
  to = options.delete(:to)
  if args.empty? && !to
    if options.length == 1
      self.origins= options.keys[0]
      self.targets= options.values[0]
    else
      raise options.inspect
    end
  else
    self.origins= *args
    self.targets= to unless to.nil?
  end
end
