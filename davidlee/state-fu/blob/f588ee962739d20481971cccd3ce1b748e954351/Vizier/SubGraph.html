def build(lines = [], indent = 0)
  lines.map do |line|
    if line.is_a?( Array )
      build( line, indent + 1)
    else
      (" " * (indent * 4) ) + line.to_s
    end
  end.join("\n")
end

def write_comment( str, j = 0 )
  l = 40 - (j * 4)
  i = ' ' * (j * 4)
  "\n#{i}/*#{'*'*(l-2)}\n#{i}** #{ str.ljust((l - (6) - (j*4)),' ') }#{i} **\n#{i}#{'*'*(l-1)}/"
end
