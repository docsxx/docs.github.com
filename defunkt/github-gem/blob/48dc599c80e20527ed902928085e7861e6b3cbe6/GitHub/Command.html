def call(*args)
  arity = method(:command).arity
  args << nil while args.size < arity
  send :command, *args
end

def run(method, command)
  if command.is_a? Array
    command = [ 'git', command ].flatten
    GitHub.learn command.join(' ')
  else
    command = 'git ' + command
    GitHub.learn command
  end

  send method, *command
end
