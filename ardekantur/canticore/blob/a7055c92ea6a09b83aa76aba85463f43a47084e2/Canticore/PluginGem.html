def initialize name, &block
  @name = name
  @hooks = {}
  @filters = {}
  @variables = nil
  @values = VariableSet.new name

  instance_eval(&block)
  Plugin.add self
  if variables
    variables.each do |vname, data|
      Setting["#{name}.#{vname}"] = data[:default] unless Setting["#{name}.#{vname}"]
    end
  end    
end

def filter data, action
  raise RuntimeError, "don't know how to handle nested filters" unless data.size == 2
  @filters[data.first] = {} unless @filters.include? data.first
  @filters[data.first][data.last] = action
end

def haml action
  if action.to_sym == action
    output = File.read(File.join(File.dirname(caller.first), 'views', "#{action.to_s}.haml"))
  else
    output = action
  end
  Haml::Engine.new(output).def_method(PluginGem, :render, :values)
  self.render(:values => @values)
end

def variable name, hash
  @variables = {} unless @variables
  @variables[name] = hash
  @variables[name][:default] = '' unless @variables[name].include? :default
end
