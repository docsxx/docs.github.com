def generate *args
  @result = ""
  @x = Builder::XmlMarkup.new :indent => 2, :target => @result
  @x.instruct!
  @x.methodResponse {
    @x.params {
      args.each do |arg|
        @x.param {  generate_type(arg) }
      end            
    }
  }
  return @result
end

def generate_type argument
  @x.value { @x.string argument } if argument.class == String
  @x.value { @x.i4 argument } if argument.class == Fixnum
  @x.value { @x.array { @x.data { argument.each { |arg| generate_type(arg) } } } } if argument.class == Array
  @x.value { @x.struct { argument.each {|k,v| @x.member { @x.name(k.to_s); generate_type(v); } } } } if argument.class == Hash
  @x.value { @x.boolean argument ? 1 : 0 } if argument.class == TrueClass or argument.class == FalseClass
end
