def self.prepare database, theme = 'default'
  const_set(:DB, Sequel.connect("sqlite://#{File.expand_path(database)}")) unless defined?(DB)
  require 'canticore/migrations/m_1233977097'
  InitialMigration.apply(DB, :up) unless DB.table_exists? :categories

  %w[ setting category comment plugin post tag user ].each { |model| require "canticore/models/#{model}" }
  # TODO: Find a non-horrid way to migrate Sequel databases created at runtime

  # Default Settings
  Setting['blog.active_theme'] = theme unless Setting['blog.active_theme']
  Setting['blog.name'] = 'My Canticore Blog' unless Setting['blog.name']

  set :views => Theme.activate_theme(Setting['blog.active_theme'])

  return self
end

def define_metaweblog
  return if defined? @@MetaWeblog
  @@MetaWeblog = true
  
  # metaWeblog.getPost(id, user, password)
  Canticore.xmlrpc_define :metaWeblog, :getPost do |id,user,pass|
    if User.authenticated? user, pass
      post = Post[id]
      {
        :dateCreated => post.date_written,
        :userid => post.author_id,
        :postid => id,
        :description => post.body,
        :title => post.title,
        :link => "http://#{@request.env['SERVER_NAME']}/#{post.slug}",
        :permaLink => "http://#{@request.env['SERVER_NAME']}/#{post.slug}",
        :categories => []
      }
    else
      {
        :faultCode => 403,
        :faultString => "forbidden"
      }
    end
  end

end

def logged_in
  return false unless session[:canticore_user_login]
  u = Hash[*[:id, :name, :hashed_password].zip(session[:canticore_user_login].split("|")).flatten]
  return false unless User[u[:id]] # TODO: Write spec for this line -- cookie existing but user not found in DB
  return false unless u.merge!(:id => u[:id].to_i)[:hashed_password] == User[u[:id]].password
  return u
end

def partial(template, *args)
  options = args.extract_options!
  options.merge!(:layout => false)
  raise RuntimeError, "all partials currently require a :collection" unless collection = options.delete(:collection)
  collection.inject([]) do |buffer, member|
    buffer << haml("_#{template}".to_sym, options.merge(:layout => false, :locals => {template.to_sym => member}))
  end
end

def prepare_settings_for params
  @path = LayoutHandler.breadcrumb :admin, '/admin', :plugins, '/admin/plugins', params[:name].downcase, nil, :settings, nil
  @name = params.delete("name")
  @plugin = Plugin.active_plugins[@name]
  raise RuntimeError, "unable to find plugin #{@name} in loaded plugins" unless @plugin
end

def static_content root_dir, params
  whitelist = %w/ jpg jpeg png gif css html txt /
  throw :halt, [403, 'forbidden'] unless whitelist.include? params[:ext]

  content_type "image/#{params[:ext]}"
  content_type "image/jpeg" if %w/ jpg jpeg /.include? params[:ext]
  content_type 'text/css', :charset => 'utf-8' if params[:ext] == 'css'

  file = File.join(root_dir, 'static', "#{params[:file]}.#{params[:ext]}")

  if File.exist? file
    send_file file, :disposition => 'inline'
  else
    throw :halt, [404, 'file not found']
  end
end
