def self.activate_theme theme_name, gem_list = Gem::Specification.list
  require "canticore-theme-#{theme_name}"
  gem = gem_list.select { |gem| gem.name =~ /canticore-theme-#{theme_name}/ }.first
  raise RuntimeError, "couldn't find the right theme gem" unless gem
  Setting['blog.active_theme'] = theme_name
  Setting['blog.active_theme_path'] = gem.full_gem_path
  raise RuntimeError, "invalid theme" unless self.valid_theme? gem
  return File.join(gem.full_gem_path, 'views')
end

def self.valid_theme? gem
  return false unless gem
  REQUIRED_FILES.each do |file| 
    return false unless File.exist?(File.join(gem.full_gem_path, 'views', "#{file}.haml")) 
  end
  return true
end
